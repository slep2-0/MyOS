21// Read the FAT for the given cluster, to inspect data about this specific cluster, like which sectors are free, used, what's the next sector, and which sector are EOF (end of file = 0x0FFFFFFF)
22static uint32_t fat32_read_fat(uint32_t cluster) {
23 tracelast_func("fat32_read_fat");
24 uint32_t fat_offset = cluster * 4; // advances.
25 uint32_t fat_sector = fs.fat_start + (fat_offset / fs.bytes_per_sector);
26 uint32_t ent_offset = fat_offset % fs.bytes_per_sector;
27
28 uint32_t buf32;
29 read_sector(fat_sector, (uint8_t*)&buf32 + (ent_offset));
30 return buf32 & 0x0FFFFFFF;
31}

38// Read BPB (Bios Parameter Block) and initialize.
39bool fat32_init(int disk_index) {
40 tracelast_func("fat32_init");
41 disk = get_block_device(disk_index);
42 if (!disk) { return false; }
43
44 uint8_t buf[512];
45 if (!read_sector(0, buf)) { return false; } // First sector contains the BPB for FAT.
46 kmemcpy(&bpb, buf, sizeof(bpb)); // So copy that first sector into our local BPB structure.
47
48 // Then initialize it.
49 fs.bytes_per_sector = bpb.bytes_per_sector;
50 fs.sectors_per_cluster = bpb.sectors_per_cluster;
51 fs.reserved_sector_count = bpb.reserved_sector_count;
52 fs.sectors_per_fat = bpb.fat_size_32;
53 fs.root_cluster = bpb.root_cluster;
54
55 fs.fat_start = bpb.reserved_sector_count; // technically also reserved_sector_count of fs. holds it as well.
56 fs.first_data_sector = fs.fat_start + bpb.num_fats * fs.sectors_per_fat;
57
58 return true;
59}

83// Walk cluster chain and read directory entries.
84void fat32_list_root(void) {
85 tracelast_func("fat32_list_root");
86 uint32_t cluster = fs.root_cluster;
87 do {
88 uint32_t sector = first_sector_of_cluster(cluster);
89 for (uint32_t i = 0; i < fs.sectors_per_cluster; i++) {
90 uint8_t buf[512];
91 if (!read_sector(sector + i, buf)) {
92 return;
93 }
94
95 FAT32_DIR_ENTRY* dir = (FAT32_DIR_ENTRY*)buf;
96
97 for (uint32_t j = 0; j < fs.bytes_per_sector / sizeof(*dir); j++, dir++) {
98 if (dir->name[0] == 0x00) {
99 return; // No more entries.
100 }
101
102 if ((uint8_t)dir->name[0] == 0xE5) {
103 continue;
104 }
105
106 if (dir->attr == ATTR_LONG_NAME) {
107 // i don't support long names for now.
108 continue;
109 }
110#ifdef DEBUG
111 gop_printf(&gop_local, 0xFF00FFFF, "Found: ");
112 for (uint32_t k = 0; k < 11; k++)
113 gop_printf(&gop_local, 0xFFD3D3D3, "%c", dir->name[k]);
114 gop_printf(&gop_local, 0, "\n");
115#endif
116 }
117 }
118 cluster = fat32_read_fat(cluster);
119 } while (cluster < 0x0FFFFFF8);
120}

122// Read a file into buffer.
123bool fat32_read_file(const char* filename, void* buffer, uint32_t buffer_size) {
124 tracelast_func("fat32_read_file");
125 uint32_t cluster = fs.root_cluster;
126
127 do {
128 uint32_t sector = first_sector_of_cluster(cluster);
129 for (uint32_t i = 0; i < fs.sectors_per_cluster; i++) {
130 uint8_t sblk[512];
131 if (!read_sector(sector + i, sblk)) {
132 return false;
133 }
134
135 FAT32_DIR_ENTRY* dir = (FAT32_DIR_ENTRY*)sblk;
136
137 for (uint32_t j = 0; j < fs.bytes_per_sector / sizeof(*dir); j++, dir++) {
138 if (dir->name[0] == 0x00) {
139 return false; // No more entries
140 }
141
142 if (dir->attr == ATTR_LONG_NAME) {
143 continue;
144 }
145
146 if (cmp_name(dir->name, filename)) {
147 uint32_t file_cluster = (uint32_t)((dir->fst_clus_hi << 16) | dir->fst_clus_lo);
148 uint32_t remain = dir->file_size;
149 uint8_t* dst = (uint8_t*)buffer;
150
151 // Read file data cluster by cluster
152 while (file_cluster < 0x0FFFFFF8 && remain > 0 && buffer_size > 0) {
153 uint32_t first_sector = first_sector_of_cluster(file_cluster);
154
155 for (uint32_t sc = 0; sc < fs.sectors_per_cluster && remain > 0 && buffer_size > 0; sc++) {
156 uint8_t sector_buf[512];
157 if (!read_sector(first_sector + sc, sector_buf)) {
158 return false;
159 }
160
161 uint32_t bytes_to_copy = fs.bytes_per_sector;
162 if (bytes_to_copy > remain) bytes_to_copy = remain;
163 if (bytes_to_copy > buffer_size) bytes_to_copy = buffer_size;
164
165 // Copy only valid bytes
166 for (uint32_t b = 0; b < bytes_to_copy; b++) {
167 dst[b] = sector_buf[b];
168 }
169
170 dst += bytes_to_copy;
171 remain -= bytes_to_copy;
172 buffer_size -= bytes_to_copy;
173 }
174 file_cluster = fat32_read_fat(file_cluster);
175 }
176 return true;
177 }
178 }
179 }
180 cluster = fat32_read_fat(cluster);
181 } while (cluster < 0x0FFFFFF8);
182
183 return false;
184}

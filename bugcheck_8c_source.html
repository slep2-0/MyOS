20void print_lastfunc_chain(uint32_t color) {
21 // Start at the oldest entry: that's the slot `index` points to (next write).
22 int idx = lastfunc_history.current_index;
23
24 int start = (idx + 1) % LASTFUNC_HISTORY_SIZE;
25 bool first = true;
26 for (int i = 0; i < LASTFUNC_HISTORY_SIZE; i++) { // start from oldest to newest
27 idx = (start + i) % LASTFUNC_HISTORY_SIZE;
28 char* name = (char*)lastfunc_history.names[idx];
29 if (!*name) break;
30 if (!first) {
31 gop_printf(&gop_local, color, " -> ");
32 }
33 gop_printf(&gop_local, color, "%s", name);
34 first = false;
35 }
36}

140void bugcheck_system(CTX_FRAME* context, INT_FRAME* int_frame, BUGCHECK_CODES err_code, uint32_t additional, bool isAdditionals) {
141 // Critical system error, instead of triple faulting, we hang the system with specified error codes.
142 // Disable interrupts if they werent disabled before.
143 __cli();
144 isBugChecking = true;
145 bool isThereIntFrame = (int_frame) ? true : false; // basic ternary
146#ifdef DEBUG
147 IRQL recordedIrql = cpu.currentIrql;
148#endif
149 // Force to be redrawn from the top, instead of last place.
150 cursor_x = 0;
151 cursor_y = 0;
152 _SetIRQL(HIGH_LEVEL); // SET the irql to high level (not raise) (we could raise, but this takes less cycles and so is faster) (When I will integrate multi core functionality, this should SetIRQL to each cpu core.
153
154 // Clear the screen to blue (bsod windows style)
155 gop_clear_screen(&gop_local, 0xFF0035b8);
156 // check if nullptr deref.
157 if (err_code == PAGE_FAULT && isAdditionals && additional == 0) {
158 err_code = NULL_POINTER_DEREFERENCE;
159 }
160 // Write some debugging and an error message
161 gop_printf(&gop_local, 0xFFFFFFFF, "FATAL ERROR: Your system has encountered a fatal error.\n\n");
162 gop_printf(&gop_local, 0xFFFFFFFF, "Your system has been stopped for safety.\n\n");
163 char* stopCodeToStr = ""; // empty at first.
164 resolveStopCode(&stopCodeToStr, err_code);
165 uint64_t rspIfExist = (context->rsp) ? context->rsp : 0;
166 gop_printf(&gop_local, 0xFFFFFFFF, "**STOP CODE: ");
167 gop_printf(&gop_local, 0xFF8B0000, "%s", stopCodeToStr);
168 gop_printf(&gop_local, 0xFF00FF00, " (numerical: %d)**", err_code);
169 if (context) {
170 gop_printf(&gop_local, 0xFFFFFFFF,
171 "\n\nRegisters:\n\n"
172 "RAX: %p RBX: %p RCX: %p RDX: %p\n\n"
173 "RSI: %p RDI: %p RBP: %p\n\n"
174 "R8 : %p R9 : %p R10: %p R11: %p \n\n"
175 "R12: %p R13: %p R14: %p R15: %p\n\n\n",
176 context->rax,
177 context->rbx,
178 context->rcx,
179 context->rdx,
180 context->rsi,
181 context->rdi,
182 context->rbp,
183 rspIfExist,
184 context->r8,
185 context->r9,
186 context->r10,
187 context->r11,
188 context->r12,
189 context->r13,
190 context->r14,
191 context->r15
192 );
193 }
194 else {
195 gop_printf(&gop_local, 0xFFFF0000, "\n\n\n**ERROR: NO REGISTERS.**");
196 }
197 // don't alert if there is no interrupt frame, the user shouldn't care and know. - i should do an IFDEF here for debug, but I could not remember that I didn't define, i'd rather keep it like this for now.
198 if (isThereIntFrame) {
199 gop_printf(&gop_local, (uint32_t)-1,
200 "Exceptions:\n\n"
201 "Vector Number: %d Error Code: %p\n\n"
202 "RIP: %p CS: %p RFLAGS: %p\n",
203 int_frame->vector,
204 int_frame->error_code,
205 int_frame->rip,
206 int_frame->cs,
207 int_frame->rflags
208 );
209 }
210#ifdef DEBUG
211 gop_printf(&gop_local, 0xFFFFA500, "**Last IRQL: %d**", recordedIrql);
212#endif
213 if (isAdditionals) {
214 if (err_code == PAGE_FAULT) {
215 gop_printf(&gop_local, 0xFFFFA500, "\n\n\n**FAULTY ADDRESS: %p**", additional);
216 }
217 else {
218 gop_printf(&gop_local, 0xFFBF40BF, "\n\n\n**ADDITIONALS: %p**", additional);
219 }
220 }
221#ifdef DEBUG
222 if (lastfunc_history.names[lastfunc_history.current_index][0] != '\0') {
223 gop_printf(&gop_local, 0xFFBF40BF, "\n\n**FUNCTION TRACE (oldest to newest): ");
224 print_lastfunc_chain(0xFFBF40BF);
225 gop_printf(&gop_local, 0xFFBF40BF, "**");
226 }
227#endif
228 //test
229 update_pic_mask_for_current_irql();
230 __cli();
231 // spin the CPU.
232 while (1) {
233 NOTHING;
234 }
235}

110static inline uint16_t fat32_encode_date(uint16_t year, uint8_t month, uint8_t day) {
111 // FAT date: YYYYYYYMMMMDDDDD (since 1980)
112 return ((year - 1980) << 9) | (month << 5) | (day);
113}
114
115static inline uint16_t fat32_encode_time(uint8_t hour, uint8_t min, uint8_t sec) {
116 // FAT time: HHHHHMMMMMMSSSSS (seconds / 2)
117 return (hour << 11) | (min << 5) | (sec / 2);
118}

120static inline void fat32_decode_date(uint16_t date, uint16_t* year, uint8_t* month, uint8_t* day) {
121 *year = 1980 + ((date >> 9) & 0x7F); // 7 bits for year
122 *month = (date >> 5) & 0x0F; // 4 bits for month
123 *day = date & 0x1F; // 5 bits for day
124}

126static inline void fat32_decode_time(uint16_t time, uint8_t* hour, uint8_t* min, uint8_t* sec) {
127 *hour = (time >> 11) & 0x1F; // 5 bits for hours
128 *min = (time >> 5) & 0x3F; // 6 bits for minutes
129 *sec = (time & 0x1F) * 2; // 5 bits for seconds (2-second resolution)
130}

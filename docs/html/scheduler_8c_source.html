23void InitScheduler(void) {
24 tracelast_func("InitScheduler");
25 cpu.schedulerEnabled = true;
26
27 CTX_FRAME cfm;
28 kmemset(&cfm, 0, sizeof(cfm)); // Start with a clean, all-zero context
29
30 // Set only the essential registers for starting the thread
31 cfm.rsp = (uint64_t)(idleStack + IDLE_STACK_SIZE);
32 cfm.rip = (uint64_t)kernel_idle_checks;
33
34 // Assign the clean context to the idle thread
35 idleThread.registers = cfm;
36 idleThread.threadState = READY;
37 idleThread.nextThread = NULL;
38 idleThread.TID = 0; // Scheduler thread, TID is 0.
39
40 cpu.currentThread = NULL;
41
42 // The ready queue starts empty
43 cpu.readyQueue.head = cpu.readyQueue.tail = NULL;
44}

46// Enqueue the thread if it's still RUNNABLE.
47static void enqueue_runnable(Thread* t) {
48 tracelast_func("enqueue_runnable");
49 if (!t) {
50 CTX_FRAME ctx;
51 SAVE_CTX_FRAME(&ctx);
52 BUGCHECK_ADDITIONALS addt;
53 kmemset((void*)&addt, 0, sizeof(BUGCHECK_ADDITIONALS));
54 ksnprintf(addt.str, sizeof(addt.str), "Thread was to be enqueued, but it is a null pointer.");
55 MtBugcheckEx(&ctx, NULL, NULL_THREAD, &addt, true);
56 }
57 if (t->threadState == RUNNING) {
58 t->threadState = READY;
59 enqueue(&cpu.readyQueue, t); // Insert into CPU ready queue
60 }
61}

63void Schedule(void) {
64 tracelast_func("Schedule");
65 IRQL oldIrql;
66 MtRaiseIRQL(DISPATCH_LEVEL, &oldIrql);
67
68 Thread* prev = cpu.currentThread;
69
70 if (prev && prev != &idleThread && prev->threadState == RUNNING) {
71 // The current thread's registers were already saved in isr_stub. (look after the pushes)
72 enqueue_runnable(prev);
73 }
74
75 Thread* next = dequeue(&cpu.readyQueue);
76
77 if (!next) {
78 next = &idleThread;
79 }
80 //gop_printf(COLOR_RED, "The thread's timeslice before change is: %d", next->timeSlice);
81 next->threadState = RUNNING;
82 next->timeSlice = next->origTimeSlice;
83 cpu.currentThread = next;
84
85 MtLowerIRQL(PASSIVE_LEVEL);
86 tracelast_func("Entering restore_context.");
87 restore_context(&next->registers);
88}

58static void insert_block_sorted(BLOCK_HEADER* newblock) {
59 tracelast_func("insert_block_sorted");
60 uint64_t rip;
61 GET_RIP(rip);
62 enforce_max_irql(DISPATCH_LEVEL, (void*)rip);
63 if (!free_list || newblock < free_list) {
64 newblock->next = free_list;
65 free_list = newblock;
66 return;
67 }
68
69 BLOCK_HEADER* current = free_list;
70 while (current->next && current->next < newblock) {
71 current = current->next;
72 }
73
74 newblock->next = current->next;
75 current->next = newblock;
76}

78/* Merge adjacent free blocks to reduce fragmentation */
79static void coalesce_free_list(void) {
80 tracelast_func("coalesce_free_list");
81 uint64_t rip;
82 GET_RIP(rip);
83 enforce_max_irql(DISPATCH_LEVEL, (void*)rip);
84 BLOCK_HEADER* b = free_list;
85 while (b && b->next) {
86 uintptr_t end_of_b = (uintptr_t)b + b->size;
87 if (end_of_b == (uintptr_t)b->next) {
88 /* Adjacent: merge */
89 b->size += b->next->size;
90 b->next = b->next->next;
91 }
92 else {
93 b = b->next;
94 }
95 }
96}

98static bool grow_heap_by_one_page(void) {
99 tracelast_func("grow_heap_by_one_page");
100 uint64_t rip;
101 GET_RIP(rip);
102 enforce_max_irql(DISPATCH_LEVEL, (void*)rip);
103 // grab a physical frame
104 uintptr_t phys = alloc_frame();
105 if (!phys) { return false; }
106
107 // map it at the end of the heap.
108 // here it where it maps the page, I was confused on how it worked, forgot about it.
109 map_page((void*)heap_current_end, phys, PAGE_PRESENT | PAGE_RW /* | PAGE_USER Would be used later on, when this kernel has both user mode and kernel mode */ );
110
111 // Zero the page.
112 kmemset((void*)heap_current_end, 0, FRAME_SIZE);
113 // insert a new region
114 BLOCK_HEADER* block = (BLOCK_HEADER*)heap_current_end;
115 block->size = FRAME_SIZE;
116 insert_block_sorted(block);
117 coalesce_free_list();
118
119 // advance our end.
120 heap_current_end += FRAME_SIZE;
121 return true;
122}

154void* MtAllocateVirtualMemory(size_t wanted_size, size_t align) {
155 tracelast_func("MtAllocateVirtualMemory");
156 uint64_t rip;
157 GET_RIP(rip);
158 enforce_max_irql(DISPATCH_LEVEL, (void*)rip);
159 // Precompute maximum pages we'll ever need (worst-case alignment overhead)
160 size_t max_overhead = sizeof(BLOCK_HEADER) + sizeof(void*) + (align - 1);
161 size_t max_alloc = max_overhead + wanted_size;
162
163 for (;;) {
164 // 1 - First fit search.
165 BLOCK_HEADER** cur = &free_list;
166 while (*cur) {
167 BLOCK_HEADER* blk = *cur;
168
169 uintptr_t raw_start = (uintptr_t)(blk + 1) + sizeof(void*);
170 uintptr_t aligned_start = (raw_start + align - 1) & ~(align - 1);
171 uintptr_t header_store = aligned_start - sizeof(void*);
172 uintptr_t alloc_end = aligned_start + wanted_size;
173 size_t real_total = alloc_end - (uintptr_t)blk;
174
175 if (blk->size >= real_total) {
176 /* If the block is big enough to split */
177 if (blk->size >= real_total + sizeof(BLOCK_HEADER) + align) {
178 /* Split off the tail into a new free block */
179 BLOCK_HEADER* next_blk = (BLOCK_HEADER*)((uintptr_t)blk + real_total);
180 next_blk->size = blk->size - real_total;
181 next_blk->next = blk->next;
182
183 /* Shrink current block to allocated size */
184 blk->size = real_total;
185 *cur = next_blk;
186 }
187 else {
188 /* Use the entire block */
189 *cur = blk->next;
190 }
191
192 // Store header pointer for free
193 blk->in_use = true;
194 blk->next = NULL;
195 blk->kind = BLK_NORMAL;
196 ((BLOCK_HEADER**)header_store)[0] = blk;
198 kmemset((void*)aligned_start, 0, wanted_size);
199 // Return the newly allocated memory ptr.
200 return (void*)aligned_start;
201 }
202
203 cur = &blk->next;
204 }
205
206 // 2 - No available block is found, grow heap conservatively
207 size_t pages_needed = (max_alloc + FRAME_SIZE - 1) / FRAME_SIZE;
208 for (size_t i = 0; i < pages_needed; i++) {
209 if (!grow_heap_by_one_page()) {
210 CTX_FRAME ctx;
211 SAVE_CTX_FRAME(&ctx);
212 MtBugcheck(&ctx, NULL, MEMORY_LIMIT_REACHED, 0, false);
213 }
214 }
215 // We have grown the amount, retry.
216 continue;
217 }
218}

227void* MtAllocateVirtualMemoryEx(size_t wanted_size, size_t align, uint64_t flags) {
228 tracelast_func("MtAllocateVirtualMemoryEx");
229 if (align == 0 || (align & (align - 1)) != 0) return NULL;
230 // 1. Calculate the total size needed, including our header and alignment padding.
231 size_t header_size = sizeof(BLOCK_HEADER) + sizeof(void*); // For header and original pointer
232 size_t total_size = wanted_size + header_size + (align - 1);
233
234 // 2. Round this up to the nearest page size (4K)
235 size_t pages_needed = (total_size + FRAME_SIZE - 1) / FRAME_SIZE;
236 size_t region_size = pages_needed * FRAME_SIZE;
237
238 // 3. Map a new, contiguous region of memory with the specified flags.
239 // We will map it at the end of the current heap.
240 void* region_start_virt = (void*)heap_current_end;
241
242 for (size_t i = 0; i < pages_needed; i++) {
243 uintptr_t phys = alloc_frame();
244 if (!phys) {
245 // Out of memory, we should probably shrink heap_current_end back down and fail.
246 // For now, we'll bugcheck.
247 CTX_FRAME ctx;
248 SAVE_CTX_FRAME(&ctx);
249 MtBugcheck(&ctx, NULL, MEMORY_LIMIT_REACHED, 0, false);
250 }
251 void* va = (uint8_t*)region_start_virt + (i * FRAME_SIZE);
252 map_page(va, phys, flags);
253 }
254
255 heap_current_end += region_size; // Claim the virtual address space immediately
256
257 // 4. Set up the block header at the start of our new region.
258 BLOCK_HEADER* blk = (BLOCK_HEADER*)region_start_virt;
259 blk->size = region_size;
260 blk->next = NULL;
261 blk->in_use = true;
262 blk->kind = BLK_EX;
263
264 // 5. Perform the same alignment logic as before to get the final user pointer.
265 uintptr_t raw_start = (uintptr_t)(blk + 1) + sizeof(void*);
266 uintptr_t aligned_start = (raw_start + align - 1) & ~(align - 1);
267 uintptr_t header_store = aligned_start - sizeof(void*);
268
269 // Store the original block pointer for the free function.
270 ((BLOCK_HEADER**)header_store)[0] = blk;
271
272 kmemset((void*)aligned_start, 0, wanted_size);
273 return (void*)aligned_start;
274}

276void MtFreeVirtualMemory(void* ptr) {
277 if (!ptr) return;
278 tracelast_func("MtFreeVirtualMemory");
279
280 // Get the block header
281 BLOCK_HEADER* blk = ((BLOCK_HEADER**)ptr)[-1];
282
283 // Check for our magic number
284 if (blk->kind == BLK_EX) {
285 // This is a special allocation from MtAllocateVirtualMemoryEx.
286 // We need to unmap its pages instead of returning it to the free list.
287 size_t region_size = blk->size;
288 size_t pages_to_unmap = region_size / FRAME_SIZE;
289 blk->in_use = false;
290 blk->kind = 0;
291 for (size_t i = 0; i < pages_to_unmap; i++) {
292 void* va_to_unmap = (uint8_t*)blk + (i * FRAME_SIZE);
293 unmap_page(va_to_unmap); // This also frees the physical frame
294 }
295 return;
296 }
297
298 kmemset((void*)(blk + 1), 0, blk->size - sizeof(BLOCK_HEADER));
299 blk->in_use = false;
300 insert_block_sorted(blk);
301 coalesce_free_list();
302}

20static inline bool interrupts_enabled(void) {
21 unsigned long flags;
22 __asm__ __volatile__("pushfq; popq %0" : "=r"(flags));
23 return (flags & (1UL << 9)) != 0; // IF is bit 9
24}

26// IMPORTANT: We disable interrupts around PIC updates to avoid races.
27void update_pic_mask_for_current_irql(void) {
28 bool prev_if = interrupts_enabled();
29 __cli();
30 IRQL level = cpu.currentIrql;
31
32 // Mask any IRQ whose assigned IRQL is <= the current CPU IRQL.
33 // Unmask any IRQ whose assigned IRQL is > the current CPU IRQL.
34 for (uint8_t i = 0; i < IRQ_LINES; i++) {
35 if (irq_irql[i] <= level) {
36 mask_irq(i);
37 }
38 else {
39 unmask_irq(i);
40 }
41 }
42 if (prev_if) __sti();
43}

55void MtRaiseIRQL(IRQL new_irql, IRQL* old_irql) {
56 bool prev_if = interrupts_enabled();
57 __cli();
58 tracelast_func("RaiseIRQL");
59
60 if (old_irql) {
61 *old_irql = cpu.currentIrql;
62 }
63
64 IRQL curr = atomic_load_explicit(&cpu.currentIrql, memory_order_acquire);
65 if (new_irql < curr) {
66 CTX_FRAME ctx;
67 SAVE_CTX_FRAME(&ctx);
68 BUGCHECK_ADDITIONALS addt = { 0 };
69 ksnprintf(addt.str, sizeof(addt.str), "Attempted to raise IRQL to a lower level than current IRQL.");
70 MtBugcheckEx(&ctx, NULL, IRQL_NOT_LESS_OR_EQUAL, &addt, true);
71 }
72
73 cpu.currentIrql = new_irql;
74 toggle_scheduler();
75 update_pic_mask_for_current_irql();
76 if (prev_if) __sti();
77}

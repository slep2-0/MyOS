19static inline uint64_t save_and_cli(void) {
20 uint64_t flags;
21 __asm__ volatile (
22 "pushfq\n\t"
23 "popq %0\n\t"
24 "cli"
25 : "=r"(flags)
26 :: "memory"
27 );
28 return flags;
29}

53static inline void MtAcquireSpinlock(SPINLOCK* lock, uint64_t* flags_out) {
54 tracelast_func("MtAcquireSpinlock");
55 if (!lock) return;
56 // spin until we grab the lock.
57 *flags_out = save_and_cli();
58 while (atomic_flag_test_and_set_explicit(&lock->LOCKED, memory_order_acquire)) {
59 //__asm__ volatile ("pause");

70 if (!lock) return;

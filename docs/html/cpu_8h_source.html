117static inline void enqueue(Queue* queue, Thread* thread) {
118 tracelast_func("enqueue");
119 uint64_t flags;
120 MtAcquireSpinlock(&queueLock, &flags);
121 thread->nextThread = NULL;
122 if (!queue->head) queue->head = thread;
123 else queue->tail->nextThread = thread;
124 queue->tail = thread;
125 MtReleaseSpinlock(&queueLock, flags);
126}
127
128static inline Thread* dequeue(Queue* q) {
129 tracelast_func("dequeue");
130 uint64_t flags;
131 MtAcquireSpinlock(&queueLock, &flags);
132 if (!q->head) {
133 return NULL;
134 }
135
136 Thread* t = q->head;
137 q->head = t->nextThread;
138 if (!q->head) {
139 q->tail = NULL;
140 }
141 t->nextThread = NULL;
142 MtReleaseSpinlock(&queueLock, flags);
143 return t;
144}

40// Compute checksum of 8.3 name (from specification)
41static uint8_t lfn_checksum(const uint8_t short_name[11]) {
42 uint8_t sum = 0;
43 for (int i = 0; i < 11; i++) {
44 sum = ((sum & 1) ? 0x80 : 0) + (sum >> 1) + short_name[i];
45 }
46 return sum;
47}

72// Helper: convert "NAME.EXT" or "NAMEEXT" to 11-byte FAT short-name (uppercased, space-padded).
73static void format_short_name(const char* input, char out[11]) {
74 // Fill with spaces
75 for (int i = 0; i < 11; ++i) out[i] = ' ';
76 // Copy name (up to 8 chars)
77 int ni = 0;
78 const unsigned char* p = (const unsigned char*)input;
79 while (*p && *p != '.' && ni < 8) {
80 out[ni++] = (char)toupper(*p++);
81 }
82 // If '.' found, copy extension (up to 3)
83 if (*p == '.') {
84 ++p;
85 int ei = 0;
86 while (*p && ei < 3) {
87 out[8 + ei++] = (char)toupper(*p++);
88 }
89 }
90}

99static FAT32_DIR_ENTRY* read_lfn(FAT32_DIR_ENTRY* cur, uint32_t remaining, char* out_name, uint32_t* out_consumed) {
100 if (!cur || remaining == 0) return NULL;
101 *out_consumed = 0;
102
103 // Collect LFN pointers (they appear immediately before the 8.3 entry).
104 FAT32_DIR_ENTRY* lfn_list[MAX_LFN_ENTRIES];
105 uint32_t lfn_count = 0;
106 uint32_t i = 0;
107
108 // Walk forward while entries are LFN (0x0F). Stop when we hit a non-LFN or end.
109 while (i < remaining && ((uint8_t)cur[i].name[0] != 0x00) && (cur[i].attr == ATTR_LONG_NAME)) {
110 if (lfn_count < MAX_LFN_ENTRIES) lfn_list[lfn_count++] = &cur[i];
111 i++;
112 }
113
114 // i now points to the candidate 8.3 entry (must exist and not be end marker).
115 if (i >= remaining) return NULL;
116 FAT32_DIR_ENTRY* short_entry = &cur[i];
117 if ((uint8_t)short_entry->name[0] == 0x00 || (uint8_t)short_entry->name[0] == 0xE5) {
118 // no valid 8.3 here
119 return NULL;
120 }
121
122 // If no LFN entries collected, just copy short name into out_name and return short_entry.
123 if (lfn_count == 0) {
124 // Convert 11-byte SFN to human readable name "NAME.EXT"
125 const unsigned char* s = (const unsigned char*)short_entry->name;
126 int pos = 0;
127
128 // name part (0..7)
129 for (int n = 0; n < 8; ++n) {
130 if (s[n] == ' ') break;
131 if (pos < (int)MAX_LFN_LEN - 1) out_name[pos++] = s[n];
132 }
133
134 // extension (8..10)
135 bool has_ext = false;
136 for (int n = 8; n < 11; ++n) if (s[n] != ' ') { has_ext = true; break; }
137 if (has_ext) {
138 if (pos < (int)MAX_LFN_LEN - 1) out_name[pos++] = '.';
139 for (int n = 8; n < 11; ++n) {
140 if (s[n] == ' ') break;
141 if (pos < (int)MAX_LFN_LEN - 1) out_name[pos++] = s[n];
142 }
143 }
144
145 out_name[pos] = '\0';
146 *out_consumed = 1;
147 return short_entry;
148 }
149
150 // Validate checksum of short name against each LFN entry's checksum field (offset 13)
151 uint8_t cs = lfn_checksum((uint8_t*)short_entry->name);
152 for (uint32_t j = 0; j < lfn_count; ++j) {
153 uint8_t entry_checksum = *((uint8_t*)lfn_list[j] + 13);
154 if (entry_checksum != cs) return NULL; // mismatch -> invalid chain
155 }
156
157 // Reconstruct name: iterate lfn_list in reverse (last chunk -> first chunk)
158 uint32_t pos = 0;
159 for (int j = (int)lfn_count - 1; j >= 0; --j) {
160 uint8_t* ebytes = (uint8_t*)lfn_list[j];
161
162 // Name1 at offset 1, 5 UTF-16 chars
163 uint16_t* name1 = (uint16_t*)(ebytes + 1);
164 for (int c = 0; c < 5; ++c) {
165 uint16_t ch = name1[c];
166 if (ch == 0x0000) { out_name[pos] = '\0'; goto done; }
167 if (ch <= 0x7F) out_name[pos++] = (char)ch; else out_name[pos++] = '?';
168 if (pos >= MAX_LFN_LEN - 1) goto done;
169 }
170
171 // Name2 at offset 14, 6 UTF-16 chars
172 uint16_t* name2 = (uint16_t*)(ebytes + 14);
173 for (int c = 0; c < 6; ++c) {
174 uint16_t ch = name2[c];
175 if (ch == 0x0000) { out_name[pos] = '\0'; goto done; }
176 if (ch <= 0x7F) out_name[pos++] = (char)ch; else out_name[pos++] = '?';
177 if (pos >= MAX_LFN_LEN - 1) goto done;
178 }
179
180 // Name3 at offset 28, 2 UTF-16 chars
181 uint16_t* name3 = (uint16_t*)(ebytes + 28);
182 for (int c = 0; c < 2; ++c) {
183 uint16_t ch = name3[c];
184 if (ch == 0x0000) { out_name[pos] = '\0'; goto done; }
185 if (ch <= 0x7F) out_name[pos++] = (char)ch; else out_name[pos++] = '?';
186 if (pos >= MAX_LFN_LEN - 1) goto done;
187 }
188 }
189
190done:
191 out_name[pos] = '\0';
192 // consumed entries = number of LFN entries + the 8.3 entry
193 *out_consumed = (uint32_t)lfn_count + 1;
194 return short_entry;
195}

197// Read the FAT for the given cluster, to inspect data about this specific cluster, like which sectors are free, used, what's the next sector, and which sector are EOF (end of file = 0x0FFFFFFF)
198static uint32_t fat32_read_fat(uint32_t cluster) {
199 tracelast_func("fat32_read_fat");
200 MTSTATUS status;
201 uint32_t fat_offset = cluster * 4;
202 uint32_t fat_sector = fs.fat_start + (fat_offset / fs.bytes_per_sector);
203 uint32_t ent_offset = fat_offset % fs.bytes_per_sector;
204
205 void* buf = MtAllocateVirtualMemory(512, 512);
206 if (!buf) return 0x0FFFFFFF; // treat as EOF on allocation failure
207 status = read_sector(fat_sector, buf);
208 if (MT_FAILURE(status)) {
209 tracelast_func("Couldn't read sector.");
210 MtFreeVirtualMemory(buf);
211 return 0x0FFFFFFF;
212 }
213
214 uint32_t val = *(uint32_t*)((uint8_t*)buf + ent_offset);
215 val &= 0x0FFFFFFF;
216 MtFreeVirtualMemory(buf);
217 return val;
218}

226static bool fat32_write_fat(uint32_t cluster, uint32_t value) {
227 uint32_t fat_offset = cluster * 4;
228 uint32_t fat_sector_base = fs.fat_start; // first FAT start
229 uint32_t ent_offset = fat_offset % fs.bytes_per_sector;
230 uint32_t sec_index = fat_offset / fs.bytes_per_sector;
231
232 void* buf = MtAllocateVirtualMemory(512, 512);
233 if (!buf) return false;
234 MTSTATUS status;
235 bool ok = true;
236 for (uint32_t fat_i = 0; fat_i < bpb.num_fats; ++fat_i) {
237 uint32_t fat_sector = fat_sector_base + fat_i * fs.sectors_per_fat + sec_index;
238 status = read_sector(fat_sector, buf);
239 if (MT_FAILURE(status)) { ok = false; break; }
240
241 uint32_t* fat_entry = (uint32_t*)((uint8_t*)buf + ent_offset);
242 *fat_entry = (*fat_entry & 0xF0000000) | (value & 0x0FFFFFFF);
243 status = write_sector(fat_sector, buf);
244 if (MT_FAILURE(status)) { ok = false; break; }
245 }
246
247 MtFreeVirtualMemory(buf);
248 return ok;
249}

256// Free a cluster chain starting at start_cluster (set each entry to FREE)
257static bool fat32_free_cluster_chain(uint32_t start_cluster) {
258 tracelast_func("fat32_free_cluster_chain");
259 if (start_cluster < 2 || start_cluster >= FAT32_EOC_MIN) return false;
260
261 uint32_t cur = start_cluster;
262 while (cur < FAT32_EOC_MIN) {
263 uint32_t next = fat32_read_fat(cur);
264 // mark current as free
265 if (!fat32_write_fat(cur, FAT32_FREE_CLUSTER)) return false;
266 // protect against pathological loops
267 if (next == cur) break;
268 cur = next;
269 }
270 return true;
271}

275 tracelast_func("fat32_find_free_cluster");
276 // Start searching from cluster 2 (the first usable cluster)
277 // In a more advanced implementation, we would use the FSInfo sector to find a hint. But even then that hint can be misleading (read osdev on FAT)
278 uint32_t total_clusters = (bpb.total_sectors_32 - fs.first_data_sector) / fs.sectors_per_cluster;
279 for (uint32_t i = 2; i < total_clusters; i++) {
280 if (fat32_read_fat(i) == FAT32_FREE_CLUSTER) {
281 return i;
282 }
283 }
284 return 0; // no free clusters found..
285}

332static bool fat32_find_entry(const char* path, FAT32_DIR_ENTRY* out_entry, uint32_t* out_parent_cluster) {
333 char path_copy[260];
334 kstrcpy(path_copy, path); // Use a mutable copy
335 MTSTATUS status;
336 uint32_t current_cluster = fs.root_cluster;
337 uint32_t parent_cluster_of_last_found = fs.root_cluster;
338
339 // Handle the root path "/" separately.
340 if (kstrcmp(path_copy, "/") == 0 || path_copy[0] == '\0') {
341 if (out_entry) {
342 kmemset(out_entry, 0, sizeof(FAT32_DIR_ENTRY));
343 out_entry->attr = ATTR_DIRECTORY;
344 out_entry->fst_clus_lo = (uint16_t)(fs.root_cluster & 0xFFFF);
345 out_entry->fst_clus_hi = (uint16_t)(fs.root_cluster >> 16);
346 }
347 if (out_parent_cluster) *out_parent_cluster = fs.root_cluster; // Root's parent is itself
348 return true;
349 }
350
351 FAT32_DIR_ENTRY last_found_entry;
352 kmemset(&last_found_entry, 0, sizeof(FAT32_DIR_ENTRY));
353 bool any_token_found = false;
354
355 char* token = kstrtok(path_copy, "/");
356
357 while (token != NULL) {
358 bool found_this_token = false;
359 parent_cluster_of_last_found = current_cluster;
360
361 void* sector_buf = MtAllocateVirtualMemory(512, 512);
362 if (!sector_buf) return false;
363
364 // Search the current directory (current_cluster) for the token
365 do {
366 uint32_t sector = first_sector_of_cluster(current_cluster);
367 for (uint32_t i = 0; i < fs.sectors_per_cluster; i++) {
368 status = read_sector(sector + i, sector_buf);
369 if (MT_FAILURE(status)) { MtFreeVirtualMemory(sector_buf); return false; }
370
371 FAT32_DIR_ENTRY* entries = (FAT32_DIR_ENTRY*)sector_buf;
372 uint32_t num_entries = fs.bytes_per_sector / sizeof(FAT32_DIR_ENTRY);
373
374 for (uint32_t j = 0; j < num_entries; ) {
375 if (entries[j].name[0] == END_OF_DIRECTORY) goto next_cluster_search;
376 if ((uint8_t)entries[j].name[0] == DELETED_DIR_ENTRY) { j++; continue; }
377
378 char lfn_buf[MAX_LFN_LEN];
379 uint32_t consumed = 0;
380 FAT32_DIR_ENTRY* sfn = read_lfn(&entries[j], num_entries - j, lfn_buf, &consumed);
381
382 if (sfn) {
383 bool match = false;
384
385 // 1) exact LFN match
386 if (kstrcmp(lfn_buf, token) == 0) match = true;
387
388 // 2) case-insensitive LFN match
389 if (!match) {
390 size_t la = kstrlen(lfn_buf);
391 size_t lb = kstrlen(token);
392 if (la == lb) {
393 bool eq = true;
394 for (size_t z = 0; z < la; ++z) {
395 char ca = lfn_buf[z];
396 char cb = token[z];
397 if (ca >= 'a' && ca <= 'z') ca -= ('a' - 'A');
398 if (cb >= 'a' && cb <= 'z') cb -= ('a' - 'A');
399 if (ca != cb) { eq = false; break; }
400 }
401 if (eq) match = true;
402 }
403 }
404
405 // 3) SFN compare (format token to 11-byte SFN and compare bytes)
406 if (!match) {
407 char token_sfn[11];
408 format_short_name(token, token_sfn);
409 if (cmp_short_name(sfn->name, token_sfn)) match = true;
410 }
411
412 if (match) {
413 kmemcpy(&last_found_entry, sfn, sizeof(FAT32_DIR_ENTRY));
414 found_this_token = true;
415 current_cluster = (sfn->fst_clus_hi << 16) | sfn->fst_clus_lo;
416 goto token_found_and_continue;
417 }
418 }
419
420 j += (consumed > 0) ? consumed : 1;
421 }
422 }
423 next_cluster_search:
424 current_cluster = fat32_read_fat(current_cluster);
425 } while (current_cluster < FAT32_EOC_MIN);
426
427 token_found_and_continue:
428 // free sector_buf
429 MtFreeVirtualMemory(sector_buf);
430 if (!found_this_token) {
431 return false; // Path component not found, so the whole path is invalid.
432 }
433
434 any_token_found = true;
435
436 // Get the next part of the path
437 token = kstrtok(NULL, "/");
438
439 // If we found a component, but it's not a directory and there's more path to parse, it's an error.
440 if (token != NULL && !(last_found_entry.attr & ATTR_DIRECTORY)) {
441 return false;
442 }
443 }
444
445 if (any_token_found) {
446 if (out_entry) kmemcpy(out_entry, &last_found_entry, sizeof(FAT32_DIR_ENTRY));
447 if (out_parent_cluster) *out_parent_cluster = parent_cluster_of_last_found;
448 return true;
449 }
450
451 return false; // Should not be reached if path is not root, but good for safety.
452}

454static bool fat32_extend_directory(uint32_t dir_cluster) {
455 uint32_t new_cluster = fat32_find_free_cluster();
456 if (new_cluster == 0) return false;
457
458 // Zero out the new cluster
459 if (!zero_cluster(new_cluster)) {
460 fat32_write_fat(new_cluster, FAT32_FREE_CLUSTER); // Free it back
461 return false;
462 }
463
464 fat32_write_fat(new_cluster, FAT32_EOC_MAX); // Mark as new end of chain
465
466 // Find the last cluster in the original chain and link it to the new one
467 uint32_t current = dir_cluster;
468 uint32_t next = 0;
469 while ((next = fat32_read_fat(current)) < FAT32_EOC_MIN) {
470 current = next;
471 }
472
473 return fat32_write_fat(current, new_cluster);
474}

476static bool fat32_find_free_dir_slots(uint32_t dir_cluster, uint32_t count, uint32_t* out_sector, uint32_t* out_entry_index) {
477 uint32_t current_cluster = dir_cluster;
478 void* sector_buf = MtAllocateVirtualMemory(512, 512);
479 if (!sector_buf) return false;
480 MTSTATUS status;
481 do {
482 uint32_t sector_lba = first_sector_of_cluster(current_cluster);
483 for (uint32_t i = 0; i < fs.sectors_per_cluster; i++) {
484 status = read_sector(sector_lba + i, sector_buf);
485 if (MT_FAILURE(status)) { MtFreeVirtualMemory(sector_buf); return false; }
486
487 FAT32_DIR_ENTRY* entries = (FAT32_DIR_ENTRY*)sector_buf;
488 uint32_t num_entries = fs.bytes_per_sector / sizeof(FAT32_DIR_ENTRY);
489
490 uint32_t consecutive_free = 0;
491 for (uint32_t j = 0; j < num_entries; j++) {
492 uint8_t first_byte = (uint8_t)entries[j].name[0];
493 if (first_byte == END_OF_DIRECTORY || first_byte == DELETED_DIR_ENTRY) {
494 consecutive_free++;
495 if (consecutive_free == count) {
496 *out_sector = sector_lba + i;
497 *out_entry_index = j - (count - 1);
498 MtFreeVirtualMemory(sector_buf);
499 return true;
500 }
501 }
502 else {
503 consecutive_free = 0;
504 }
505 }
506 }
507
508 uint32_t next_cluster = fat32_read_fat(current_cluster);
509 if (next_cluster >= FAT32_EOC_MIN) {
510 // End of chain, no space found, try to extend
511 if (fat32_extend_directory(dir_cluster)) {
512 // After extending, restart the search. A more optimized way exists,
513 // but this is safer and simpler to implement.
514 return fat32_find_free_dir_slots(dir_cluster, count, out_sector, out_entry_index);
515 }
516 else {
517 MtFreeVirtualMemory(sector_buf);
518 return false; // Cannot extend directory
519 }
520 }
521 current_cluster = next_cluster;
522 } while (true);
523
524 MtFreeVirtualMemory(sector_buf);
525 return false;
526}

530// Read BPB (Bios Parameter Block) and initialize.
531MTSTATUS fat32_init(int disk_index) {
532 tracelast_func("fat32_init");
533 MTSTATUS status;
534 disk = get_block_device(disk_index);
535 if (!disk) { return MT_GENERAL_FAILURE; }
536
537 void* buf = MtAllocateVirtualMemory(512, 512);
538 if (!buf) return MT_NO_MEMORY;
539 status = read_sector(BPB_SECTOR_START, buf);
540 if (MT_FAILURE(status)) { return status; } // First sector contains the BPB for FAT.
541 kmemcpy(&bpb, buf, sizeof(bpb)); // So copy that first sector into our local BPB structure.
542
543 // Then initialize it.
544 fs.bytes_per_sector = bpb.bytes_per_sector;
545 fs.sectors_per_cluster = bpb.sectors_per_cluster;
546 fs.reserved_sector_count = bpb.reserved_sector_count;
547 fs.sectors_per_fat = bpb.fat_size_32;
548 fs.root_cluster = bpb.root_cluster;
549
550 fs.fat_start = BPB_SECTOR_START + bpb.reserved_sector_count; // technically also reserved_sector_count of fs. holds it as well.
551 fs.first_data_sector = fs.fat_start + bpb.num_fats * fs.sectors_per_fat;
552 MtFreeVirtualMemory(buf);
553 return MT_SUCCESS;
554}

556// Walk cluster chain and read directory entries.
557void fat32_list_root(void) {
558 tracelast_func("fat32_list_root");
559 uint32_t cluster = fs.root_cluster;
560
561 void* buf = MtAllocateVirtualMemory(512, 512);
562 if (!buf) return;
563
564 // Temp buffer to accumulate LFN entries (and eventually the 8.3 entry).
565 FAT32_DIR_ENTRY temp_entries[MAX_LFN_ENTRIES + 1];
566 uint32_t lfn_accum = 0;
567 MTSTATUS status;
568 do {
569 uint32_t sector = first_sector_of_cluster(cluster);
570 for (uint32_t i = 0; i < fs.sectors_per_cluster; ++i) {
571 status = read_sector(sector + i, buf);
572 if (MT_FAILURE(status)) return;
573
574 FAT32_DIR_ENTRY* dir = (FAT32_DIR_ENTRY*)buf;
575 uint32_t entries = fs.bytes_per_sector / sizeof(*dir);
576
577 for (uint32_t j = 0; j < entries; ++j, ++dir) {
578 uint8_t first = (uint8_t)dir->name[0];
579
580 // End of directory: stop everything
581 if (first == 0x00) {
582 MtFreeVirtualMemory(buf);
583 return;
584 }
585
586 // Deleted entry: if we were accumulating an LFN chain, drop it.
587 if (first == 0xE5) {
588 lfn_accum = 0;
589 continue;
590 }
591
592 // If it's an LFN entry, copy it into the temp accumulator (preserve order read-on-disk)
593 if (dir->attr == ATTR_LONG_NAME) {
594 if (lfn_accum < MAX_LFN_ENTRIES) {
595 kmemcpy(&temp_entries[lfn_accum], dir, sizeof(FAT32_DIR_ENTRY));
596 lfn_accum++;
597 }
598 else {
599 // too many parts: drop accumulator to avoid overflow
600 lfn_accum = 0;
601 }
602 continue;
603 }
604 // Non-LFN entry: this is the 8.3 entry that ends any preceding LFN chain (if present).
605 // If we have accumulated LFN entries, build a contiguous array: [LFN...][8.3]
606 char bufferLfn[MAX_LFN_LEN];
607 uint32_t consumed = 0;
608 FAT32_DIR_ENTRY* real = NULL;
609
610 if (lfn_accum > 0) {
611 // copy the 8.3 entry as the last element
612 kmemcpy(&temp_entries[lfn_accum], dir, sizeof(FAT32_DIR_ENTRY));
613 // call read_lfn on our temp buffer (which starts with LFN entries)
614 real = read_lfn(temp_entries, lfn_accum + 1, bufferLfn, &consumed);
615 // reset accumulator regardless (we've handled or attempted to)
616 lfn_accum = 0;
617 }
618 else {
619 // No accumulated LFN entries: handle short-name-only entry
620 // We can call read_lfn directly on the sector buffer at current position
621 uint32_t remaining = entries - j;
622 real = read_lfn(&dir[0], remaining, bufferLfn, &consumed);
623 // note: consumed will be at least 1 if successful
624 }
625
626 // If read_lfn returned a real 8.3 entry (it should), print the name
627 if (real) {
628 gop_printf(0xFF00FFFF, "Found: %s\n", bufferLfn);
629 }
630 else {
631 // Fallback: if read_lfn failed for some reason, print raw 8.3 as a last resort
632 char fallback[12];
633 for (int k = 0; k < 11; ++k) fallback[k] = dir->name[k];
634 fallback[11] = '\0';
635 gop_printf(0xFF00FFFF, "Found (raw): %s\n", fallback);
636 }
637
638 // continue to next entry (we already advanced j by loop)
639 } // for each dir entry in sector
640 } // for each sector in cluster
641
642 cluster = fat32_read_fat(cluster);
643 } while (cluster < 0x0FFFFFF8);
644}

646// Helper to detect if a filename has a slash in it (/), and so the filename is in a directory
647static bool is_filename_in_dir(const char* filename) {
648 if (!filename) return false;
649
650 while (*filename) {
651 if (*filename == '/') return true;
652 filename++;
653 }
654
655 return false;
656}

658static uint32_t extract_dir_cluster(const char* filename) {
659 tracelast_func("extract_dir_cluster - fat32");
660
661 if (!filename || filename[0] == '\0') return fs.root_cluster;
662
663 // Make a mutable copy
664 char path_copy[260];
665 kstrcpy(path_copy, filename);
666
667 // Remove trailing slashes (keep a single leading '/' if path is "/")
668 int len = (int)kstrlen(path_copy);
669 while (len > 1 && path_copy[len - 1] == '/') {
670 path_copy[len - 1] = '\0';
671 len--;
672 }
673
674 // Find last slash
675 int last_slash = -1;
676 for (int i = len - 1; i >= 0; --i) {
677 if (path_copy[i] == '/') { last_slash = i; break; }
678 }
679
680 // No slash -> file is in root
681 if (last_slash == -1) {
682 return fs.root_cluster;
683 }
684
685 // Parent path is "/" if last_slash == 0, otherwise substring [0..last_slash-1]
686 char parent[260];
687 if (last_slash == 0) {
688 parent[0] = '/';
689 parent[1] = '\0';
690 }
691 else {
692 // copy up to last_slash
693 for (int i = 0; i < last_slash; ++i) parent[i] = path_copy[i];
694 parent[last_slash] = '\0';
695 }
696
697 // Resolve the parent path to a directory entry
698 FAT32_DIR_ENTRY parent_entry;
699 if (!fat32_find_entry(parent, &parent_entry, NULL)) return 0;
700
701 // Ensure it's a directory
702 if (!(parent_entry.attr & ATTR_DIRECTORY)) return 0;
703
704 uint32_t cluster = ((uint32_t)parent_entry.fst_clus_hi << 16) | parent_entry.fst_clus_lo;
705 if (cluster == 0) cluster = fs.root_cluster; // safety fallback
706 return cluster;
707}

709MTSTATUS fat32_read_file(const char* filename, uint32_t* file_size_out, void** buffer_out) {
710 tracelast_func("fat32_read_file");
711 MTSTATUS status;
712 // We still need a temporary buffer for reading sectors
713 void* sblk = MtAllocateVirtualMemory(512, 512);
714 if (!sblk) return MT_NO_MEMORY;
715
716 // Get the cluster of the directory filename points to (e.g "tmp/folder/myfile.txt", we need the "folder" cluster.)
717 uint32_t cluster = 0;
718
719 if (is_filename_in_dir(filename)) {
720 cluster = extract_dir_cluster(filename);
721 if (!cluster) {
722 MtFreeVirtualMemory(sblk);
723 return MT_FAT32_INVALID_CLUSTER;
724 }
725 }
726 else {
727 cluster = fs.root_cluster;
728 }
729
730 do {
731 uint32_t sector = first_sector_of_cluster(cluster);
732 for (uint32_t i = 0; i < fs.sectors_per_cluster; ++i) {
733 status = read_sector(sector + i, sblk);
734 if (MT_FAILURE(status)) {
735 // Free sblk before returning
736 MtFreeVirtualMemory(sblk);
737 return status;
738 }
739
740 FAT32_DIR_ENTRY* dir_entries = (FAT32_DIR_ENTRY*)sblk;
741 uint32_t entries_per_sector = fs.bytes_per_sector / sizeof(FAT32_DIR_ENTRY);
742
743 for (uint32_t j = 0; j < entries_per_sector; ) {
744 FAT32_DIR_ENTRY* current_entry = &dir_entries[j];
745
746 if (current_entry->name[0] == END_OF_DIRECTORY) {
747 // Free sblk
748 MtFreeVirtualMemory(sblk);
749 return MT_FAT32_FILE_NOT_FOUND; // End of directory, file not found
750 }
751 if ((uint8_t)current_entry->name[0] == DELETED_DIR_ENTRY) {
752 j++;
753 continue;
754 }
755
756 char lfn_buf[MAX_LFN_LEN];
757 uint32_t consumed_entries = 0;
758 FAT32_DIR_ENTRY* sfn_entry = read_lfn(current_entry, entries_per_sector - j, lfn_buf, &consumed_entries);
759
760 if (sfn_entry) {
761 // Check if either the long or short filename matches
762 if (kstrcmp(filename, lfn_buf) == 0) {
763 goto file_found;
764 }
765
766 char shortname_formatted[11];
767 format_short_name(filename, shortname_formatted);
768 if (cmp_short_name(sfn_entry->name, shortname_formatted)) {
769 goto file_found;
770 }
771
772 j += consumed_entries; // Skip past all LFN entries and the SFN entry
773 continue;
774
775 file_found:
776 {
777 uint32_t file_size = sfn_entry->file_size;
778 if (file_size_out) {
779 *file_size_out = file_size;
780 }
781
782 // Now allocate the final buffer for the file content
783 void* file_buffer = MtAllocateVirtualMemory(file_size, 4096);
784 if (!file_buffer) {
785 // Free sblk
786 MtFreeVirtualMemory(sblk);
787 return MT_NO_MEMORY;
788 }
789
790 uint32_t file_cluster = (uint32_t)((sfn_entry->fst_clus_hi << 16) | sfn_entry->fst_clus_lo);
791 uint32_t remaining_bytes = file_size;
792 uint8_t* dst = (uint8_t*)file_buffer;
793
794 while (file_cluster < FAT32_EOC_MIN && remaining_bytes > 0) {
795 uint32_t current_sector = first_sector_of_cluster(file_cluster);
796 for (uint32_t sc = 0; sc < fs.sectors_per_cluster && remaining_bytes > 0; ++sc) {
797 status = read_sector(current_sector + sc, sblk);
798 if (MT_FAILURE(status)) {
799 // Free both buffers
800 MtFreeVirtualMemory(file_buffer);
801 MtFreeVirtualMemory(sblk);
802 return status;
803 }
804
805 uint32_t bytes_to_copy = fs.bytes_per_sector;
806 if (bytes_to_copy > remaining_bytes) {
807 bytes_to_copy = remaining_bytes;
808 }
809
810 kmemcpy(dst, sblk, bytes_to_copy);
811 dst += bytes_to_copy;
812 remaining_bytes -= bytes_to_copy;
813 }
814 file_cluster = fat32_read_fat(file_cluster);
815 }
816
817 // Free the temporary sector buffer and return the file buffer
818 MtFreeVirtualMemory(sblk);
819 *buffer_out = file_buffer;
820 return MT_SUCCESS;
821 }
822 }
823 else {
824 j++; // Move to the next entry if read_lfn fails
825 }
826 }
827 }
828 cluster = fat32_read_fat(cluster);
829 } while (cluster < FAT32_EOC_MIN);
830
831 // Free sblk
832 MtFreeVirtualMemory(sblk);
833 return MT_FAT32_FILE_NOT_FOUND; // File not found after searching the entire directory
834}

836MTSTATUS fat32_create_directory(const char* path) {
837 tracelast_func("fat32_create_directory_full");
838 // 1. Check if an entry already exists at this path
839 if (fat32_find_entry(path, NULL, NULL)) {
840#ifdef DEBUG
841 gop_printf(0xFFFF0000, "Error: Path '%s' already exists.\n", path);
842#endif
843 return MT_FAT32_DIRECTORY_ALREADY_EXISTS;
844 }
845 MTSTATUS status;
846 // 2. Separate parent path and new directory name
847 char path_copy[260];
848 kstrcpy(path_copy, path);
849 char* new_dir_name = NULL;
850 char* parent_path = "/";
851 int last_slash = -1;
852 for (int i = 0; path_copy[i] != '\0'; i++) {
853 if (path_copy[i] == '/') last_slash = i;
854 }
855 if (last_slash != -1) {
856 new_dir_name = &path_copy[last_slash + 1];
857 if (last_slash > 0) {
858 path_copy[last_slash] = '\0';
859 parent_path = path_copy;
860 }
861 }
862 else {
863 new_dir_name = path_copy;
864 }
865
866 // 3. Find the parent directory cluster
867 FAT32_DIR_ENTRY parent_entry;
868 uint32_t parent_cluster;
869 if (!fat32_find_entry(parent_path, &parent_entry, NULL)) {
870#ifdef DEBUG
871 gop_printf(0xFFFF0000, "Error: Parent path '%s' not found.\n", parent_path);
872#endif
873 return MT_FAT32_PARENT_PATH_NOT_FOUND;
874 }
875 if (!(parent_entry.attr & ATTR_DIRECTORY)) {
876#ifdef DEBUG
877 gop_printf(0xFFFF0000, "Error: Parent path is not a directory.\n", parent_path);
878#endif
879 return MT_FAT32_PARENT_PATH_NOT_DIR;
880 }
881 parent_cluster = (parent_entry.fst_clus_hi << 16) | parent_entry.fst_clus_lo;
882
883 // 4. Allocate a new cluster for this directory's contents
884 uint32_t new_cluster = fat32_find_free_cluster();
885 if (new_cluster == 0) return MT_FAT32_CLUSTERS_FULL;
886
887 fat32_write_fat(new_cluster, FAT32_EOC_MAX);
888 zero_cluster(new_cluster);
889
890 // 5. Create '.' and '..' entries in the new cluster
891 void* sector_buf = MtAllocateVirtualMemory(512, 512);
892 if (!sector_buf) { /* handle error */ return MT_MEMORY_LIMIT; }
893 kmemset(sector_buf, 0, 512);
894 FAT32_DIR_ENTRY* dot_entries = (FAT32_DIR_ENTRY*)sector_buf;
895
896 kmemcpy(dot_entries[0].name, ". ", 11);
897 dot_entries[0].attr = ATTR_DIRECTORY;
898 dot_entries[0].fst_clus_lo = (uint16_t)new_cluster;
899 dot_entries[0].fst_clus_hi = (uint16_t)(new_cluster >> 16);
900
901 kmemcpy(dot_entries[1].name, ".. ", 11);
902 dot_entries[1].attr = ATTR_DIRECTORY;
903 dot_entries[1].fst_clus_lo = (uint16_t)parent_cluster;
904 dot_entries[1].fst_clus_hi = (uint16_t)(parent_cluster >> 16);
905
906 write_sector(first_sector_of_cluster(new_cluster), sector_buf);
907
908 // 6. Create the entry in the parent directory
909 // For simplicity, we'll use a simple SFN. A full implementation needs `fat32_generate_sfn`.
910 char sfn[11];
911 format_short_name(new_dir_name, sfn); // Using your existing simple formatter
912
913 uint32_t entry_sector, entry_index;
914 if (!fat32_find_free_dir_slots(parent_cluster, 1, &entry_sector, &entry_index)) {
915 // free sector_buf, free cluster...
916 MtFreeVirtualMemory(sector_buf);
917 fat32_write_fat(new_cluster, FAT32_FREE_CLUSTER);
918 return MT_FAT32_DIR_FULL;
919 }
920
921 // Read the target sector, modify it, write it back
922 status = read_sector(entry_sector, sector_buf);
923 if (MT_FAILURE(status)) { MtFreeVirtualMemory(sector_buf); return status; }
924
925 FAT32_DIR_ENTRY* new_entry = &((FAT32_DIR_ENTRY*)sector_buf)[entry_index];
926 kmemset(new_entry, 0, sizeof(FAT32_DIR_ENTRY));
927 kmemcpy(new_entry->name, sfn, 11);
928 new_entry->attr = ATTR_DIRECTORY;
929 new_entry->fst_clus_lo = (uint16_t)new_cluster;
930 new_entry->fst_clus_hi = (uint16_t)(new_cluster >> 16);
931
932 status = write_sector(entry_sector, sector_buf);
933
934 // free sector_buf
935 MtFreeVirtualMemory(sector_buf);
936 return status;
937}

939static uint32_t fat32_create_lfn_entries(FAT32_DIR_ENTRY* entry_buffer, const char* long_name, uint8_t sfn_checksum) {
940 uint32_t len = kstrlen(long_name);
941 uint32_t num_lfn_entries = (len + 12) / 13;
942
943 // Fill LFN entries backwards
944 for (uint32_t i = 0; i < num_lfn_entries; i++) {
945 FAT32_DIR_ENTRY* lfn = &entry_buffer[i];
946 uint32_t lfn_seq = (num_lfn_entries - i);
947 if (i == 0) lfn_seq |= 0x40; // Last LFN entry marker
948
949 lfn->attr = ATTR_LONG_NAME;
950 lfn->nt_res = 0;
951 lfn->crt_time_tenth = sfn_checksum; // This field holds the checksum
952 lfn->fst_clus_lo = 0;
953
954 // This is the sequence and type field in disguise
955 *(uint8_t*)(&lfn->name[0]) = lfn_seq;
956
957 // Character positions for LFN entry
958 // pointers into the entry (UTF-16 wide chars)
959 uint8_t* ebytes = (uint8_t*)lfn;
960 uint16_t* name1_map = (uint16_t*)(ebytes + 1); // 5 chars
961 uint16_t* name2_map = (uint16_t*)(ebytes + 14); // 6 chars
962 uint16_t* name3_map = (uint16_t*)(ebytes + 28); // 2 chars
963
964 // Populate with UTF-16 characters
965 uint32_t char_idx = (num_lfn_entries - 1 - i) * 13;
966 for (int k = 0; k < 13; k++) {
967 uint16_t uchar = 0xFFFF; // Fill with 0xFFFF
968 if (char_idx < len) {
969 uchar = long_name[char_idx];
970 }
971 else if (char_idx == len) {
972 uchar = 0x0000; // Null terminator
973 }
974
975 if (k < 5) ((uint16_t*)name1_map)[k] = uchar;
976 else if (k < 11) ((uint16_t*)name2_map)[k - 5] = uchar;
977 else ((uint16_t*)name3_map)[k - 11] = uchar;
978
979 char_idx++;
980 }
981 }
982 return num_lfn_entries;
983}

1001MTSTATUS fat32_write_file(const char* path, const void* data, uint32_t size, uint32_t mode) {
1002 tracelast_func("fat32_write_file_full");
1003 // Safety check.
1004 if (mode != WRITE_MODE_CREATE_OR_REPLACE && mode != WRITE_MODE_APPEND_EXISTING) {
1005 return MT_FAT32_INVALID_WRITE_MODE;
1006 }
1007 MTSTATUS status;
1008 uint32_t first_cluster = 0;
1009 // parse parent + filename.
1010 char path_copy[260];
1011 kstrcpy(path_copy, path);
1012 char* filename = NULL;
1013 char* parent_path = "/";
1014 int last_slash = -1;
1015 for (int i = 0; path_copy[i] != '\0'; i++) { if (path_copy[i] == '/') last_slash = i; }
1016 if (last_slash != -1) {
1017 filename = &path_copy[last_slash + 1];
1018 if (last_slash > 0) { path_copy[last_slash] = '\0'; parent_path = path_copy; }
1019 }
1020 else { filename = path_copy; }
1021
1022 // Step 3: Find parent directory cluster
1023 FAT32_DIR_ENTRY parent_entry;
1024 if (!fat32_find_entry(parent_path, &parent_entry, NULL) || !(parent_entry.attr & ATTR_DIRECTORY)) { return MT_FAT32_CLUSTER_NOT_FOUND; }
1025 uint32_t parent_cluster = (parent_entry.fst_clus_hi << 16) | parent_entry.fst_clus_lo;
1026
1027 // locate existing entry if any
1028 FAT32_DIR_ENTRY existing_entry;
1029 bool exists = fat32_find_entry(path, &existing_entry, NULL);
1030
1031 // Helper: locate on-disk sector + index for the entry (so we can update SFN/LFN in-place)
1032 uint32_t located_sector = 0;
1033 uint32_t located_index = 0;
1034 uint32_t located_consumed = 0;
1035 bool located = false;
1036
1037 {
1038 void* buf = MtAllocateVirtualMemory(512, 512);
1039 if (!buf) return MT_NO_MEMORY;
1040 uint32_t cluster = parent_cluster;
1041 do {
1042 uint32_t sector_lba = first_sector_of_cluster(cluster);
1043 for (uint32_t s = 0; s < fs.sectors_per_cluster; ++s) {
1044 status = read_sector(sector_lba + s, buf);
1045 if (MT_FAILURE(status)) { MtFreeVirtualMemory(buf); return status; }
1046 FAT32_DIR_ENTRY* entries = (FAT32_DIR_ENTRY*)buf;
1047 uint32_t entries_per_sector = fs.bytes_per_sector / sizeof(FAT32_DIR_ENTRY);
1048
1049 for (uint32_t j = 0; j < entries_per_sector; ) {
1050 uint8_t first = (uint8_t)entries[j].name[0];
1051 if (first == END_OF_DIRECTORY) { MtFreeVirtualMemory(buf); goto locate_done; }
1052 if (first == DELETED_DIR_ENTRY) { j++; continue; }
1053
1054 char lfn_buf[MAX_LFN_LEN];
1055 uint32_t consumed = 0;
1056 FAT32_DIR_ENTRY* sfn = read_lfn(&entries[j], entries_per_sector - j, lfn_buf, &consumed);
1057 if (sfn) {
1058 bool match = false;
1059 // exact LFN
1060 if (kstrcmp(lfn_buf, filename) == 0) match = true;
1061 // case-insensitive LFN
1062 if (!match && ci_equal(lfn_buf, filename)) match = true;
1063 // SFN compare
1064 if (!match) {
1065 char token_sfn[11];
1066 format_short_name(filename, token_sfn);
1067 if (cmp_short_name(sfn->name, token_sfn)) match = true;
1068 }
1069 if (match) {
1070 located_sector = sector_lba + s;
1071 located_index = j;
1072 located_consumed = consumed;
1073 located = true;
1074 MtFreeVirtualMemory(buf);
1075 goto locate_done;
1076 }
1077 j += consumed;
1078 continue;
1079 }
1080 else {
1081 j++;
1082 }
1083 }
1084 }
1085 cluster = fat32_read_fat(cluster);
1086 } while (cluster < FAT32_EOC_MIN);
1087 MtFreeVirtualMemory(buf);
1088 }
1089locate_done:
1090
1091 // 4) Allocate clusters for file data
1092 if (exists) first_cluster = (existing_entry.fst_clus_hi << 16) | existing_entry.fst_clus_lo;
1093
1094 // Behavior:
1095 // - CREATE_OR_OVERWRITE: if file exists -> free its clusters; treat as new file
1096 // - APPEND: if file exists -> append to its chain; otherwise behave like create
1097 if (mode == WRITE_MODE_CREATE_OR_REPLACE) {
1098 // If exists, free existing cluster chain and reset first_cluster
1099 if (exists && first_cluster >= 2) {
1100 if (!fat32_free_cluster_chain(first_cluster)) {
1101 return MT_FAT32_INVALID_CLUSTER;
1102 }
1103 }
1104 first_cluster = 0;
1105 }
1106 // For append, we'll compute writing start later.
1107
1108 /* Step 5: Write data to the allocated clusters */
1109 // If size > 0 allocate clusters and write data. Handle append partial-last-cluster case.
1110 if (size > 0) {
1111 uint32_t cluster_size = fs.sectors_per_cluster * fs.bytes_per_sector;
1112 uint32_t clusters_needed = 0;
1113 uint32_t prev_cluster = 0;
1114 uint32_t last_cluster = 0;
1115 uint32_t append_offset = 0;
1116
1117 if (mode == WRITE_MODE_APPEND_EXISTING && exists && first_cluster != 0) {
1118 // find last cluster of existing file and offset
1119 uint32_t cur = first_cluster;
1120 uint32_t next = cur;
1121 uint32_t file_size = existing_entry.file_size;
1122 if (file_size == 0) {
1123 last_cluster = 0;
1124 append_offset = 0;
1125 }
1126 else {
1127 // reach last cluster
1128 while (cur < FAT32_EOC_MIN) {
1129 next = fat32_read_fat(cur);
1130 if (next >= FAT32_EOC_MIN) { last_cluster = cur; break; }
1131 cur = next;
1132 }
1133 append_offset = existing_entry.file_size % cluster_size;
1134 }
1135 }
1136
1137 // compute clusters needed (account for partial last cluster when appending)
1138 if (mode == WRITE_MODE_APPEND_EXISTING && exists && append_offset > 0) {
1139 uint32_t bytes_fit = cluster_size - append_offset;
1140 if (size <= bytes_fit) clusters_needed = 0;
1141 else clusters_needed = (size - bytes_fit + cluster_size - 1) / cluster_size;
1142 }
1143 else {
1144 clusters_needed = (size + cluster_size - 1) / cluster_size;
1145 // if appending into empty file that had no clusters, clusters_needed is as above
1146 }
1147
1148 // allocate required clusters
1149 uint32_t first_new = 0;
1150 prev_cluster = 0;
1151 for (uint32_t i = 0; i < clusters_needed; ++i) {
1152 uint32_t nc = fat32_find_free_cluster();
1153 if (nc == 0) {
1154 // cleanup on failure
1155 if (first_new) fat32_free_cluster_chain(first_new);
1156 return MT_FAT32_CLUSTERS_FULL;
1157 }
1158 if (!zero_cluster(nc)) {
1159 // free and abort
1160 fat32_write_fat(nc, FAT32_FREE_CLUSTER);
1161 if (first_new) fat32_free_cluster_chain(first_new);
1162 return MT_FAT32_CLUSTER_GENERAL_FAILURE;
1163 }
1164 if (first_new == 0) first_new = nc;
1165 if (prev_cluster != 0) {
1166 if (!fat32_write_fat(prev_cluster, nc)) {
1167 // cleanup
1168 if (first_new) fat32_free_cluster_chain(first_new);
1169 return MT_FAT32_CLUSTER_GENERAL_FAILURE;
1170 }
1171 }
1172 prev_cluster = nc;
1173 }
1174 if (prev_cluster != 0) fat32_write_fat(prev_cluster, FAT32_EOC_MAX);
1175
1176 // Link newly allocated clusters into file chain
1177 if (mode == WRITE_MODE_APPEND_EXISTING && exists) {
1178 if (first_new != 0) {
1179 if (last_cluster == 0) {
1180 // file had no cluster, new allocation becomes first
1181 first_cluster = first_new;
1182 }
1183 else {
1184 // link last_cluster -> first_new
1185 if (!fat32_write_fat(last_cluster, first_new)) {
1186 if (first_new) fat32_free_cluster_chain(first_new);
1187 return MT_FAT32_CLUSTER_GENERAL_FAILURE;
1188 }
1189 }
1190 }
1191 // else nothing newly allocated, file continues as-is
1192 }
1193 else {
1194 // create/overwrite or new file -> new chain starts at first_new
1195 if (first_new != 0) first_cluster = first_new;
1196 }
1197
1198 // Now write bytes to cluster chain
1199 void* sector_buf = MtAllocateVirtualMemory(512, 512);
1200 if (!sector_buf) {
1201 // cleanup allocated clusters on error
1202 if (mode != WRITE_MODE_APPEND_EXISTING || !exists) {
1203 if (first_cluster) fat32_free_cluster_chain(first_cluster);
1204 }
1205 return MT_NO_MEMORY;
1206 }
1207
1208 const uint8_t* src = (const uint8_t*)data;
1209 uint32_t bytes_left = size;
1210
1211 // If appending into an existing partial last cluster, fill its tail first
1212 uint32_t cur_cluster = 0;
1213 uint32_t write_cluster = 0;
1214 uint32_t write_offset = 0;
1215
1216 if (mode == WRITE_MODE_APPEND_EXISTING && exists && append_offset > 0) {
1217 write_cluster = last_cluster;
1218 write_offset = append_offset;
1219 }
1220 else {
1221 write_cluster = first_cluster;
1222 write_offset = 0;
1223 }
1224
1225 // If write_cluster == 0 here (shouldn't happen when size>0) treat as error
1226 if (write_cluster == 0) {
1227 // may happen for append when file had no clusters but we didn't allocate - but in that case we did allocate above.
1228 MtFreeVirtualMemory(sector_buf);
1229 return MT_FAT32_CLUSTER_GENERAL_FAILURE;
1230 }
1231
1232 // loop writing clusters one by one
1233 cur_cluster = write_cluster;
1234 while (bytes_left > 0 && cur_cluster < FAT32_EOC_MIN) {
1235 uint32_t sector_lba = first_sector_of_cluster(cur_cluster);
1236 // write sectors in cluster
1237 for (uint32_t sc = 0; sc < fs.sectors_per_cluster && bytes_left > 0; ++sc) {
1238 // if starting in the middle of cluster, handle first sector specially
1239 if (sc == 0 && write_offset > 0) {
1240 // read sector, patch from write_offset, write back
1241 status = read_sector(sector_lba + sc, sector_buf);
1242 if (MT_FAILURE(status)) { MtFreeVirtualMemory(sector_buf); return status; }
1243 uint32_t off_in_sector = write_offset % fs.bytes_per_sector;
1244 uint32_t to_copy = fs.bytes_per_sector - off_in_sector;
1245 if (to_copy > bytes_left) to_copy = bytes_left;
1246
1247 // Copy new data
1248 kmemcpy((uint8_t*)sector_buf + off_in_sector, src, to_copy);
1249
1250 // Zero out remaining bytes in the sector
1251 if (to_copy < fs.bytes_per_sector - off_in_sector)
1252 kmemset((uint8_t*)sector_buf + off_in_sector + to_copy, 0,
1253 fs.bytes_per_sector - off_in_sector - to_copy);
1254
1255 // Patch the sector.
1256 status = write_sector(sector_lba + sc, sector_buf);
1257 if (MT_FAILURE(status)) { MtFreeVirtualMemory(sector_buf); return status; }
1258
1259 src += to_copy;
1260 bytes_left -= to_copy;
1261 write_offset = 0;
1262 continue;
1263 }
1264
1265 // normal full-sector write
1266 uint32_t to_write = (bytes_left > fs.bytes_per_sector) ? fs.bytes_per_sector : bytes_left;
1267 if (to_write < fs.bytes_per_sector) {
1268 // partial sector: read-modify-write to preserve rest
1269 status = read_sector(sector_lba + sc, sector_buf);
1270 if (MT_FAILURE(status)) { MtFreeVirtualMemory(sector_buf); return status; }
1271 kmemcpy(sector_buf, src, to_write);
1272 kmemset((uint8_t*)sector_buf + to_write, 0, fs.bytes_per_sector - to_write);
1273 status = write_sector(sector_lba + sc, sector_buf);
1274 if (MT_FAILURE(status)) { MtFreeVirtualMemory(sector_buf); return status; }
1275 }
1276 else {
1277 // full sector
1278 kmemcpy(sector_buf, src, fs.bytes_per_sector);
1279 status = write_sector(sector_lba + sc, sector_buf);
1280 if (MT_FAILURE(status)) { MtFreeVirtualMemory(sector_buf); return status; }
1281 }
1282 src += to_write;
1283 bytes_left -= to_write;
1284 }
1285 // advance to next cluster in chain
1286 uint32_t next = fat32_read_fat(cur_cluster);
1287 if (bytes_left == 0) break;
1288 if (next >= FAT32_EOC_MIN) {
1289 // Shouldn't happen because we allocated enough clusters, but guard anyway
1290 MtFreeVirtualMemory(sector_buf);
1291 return MT_FAT32_CLUSTERS_FULL;
1292 }
1293 cur_cluster = next;
1294 }
1295
1296 MtFreeVirtualMemory(sector_buf);
1297 }
1298
1299 // Step 6: Create and write directory entries (LFN + SFN)
1300 // When file already existed -> update its SFN entry in-place (file_size + fst_clus)
1301 // When file didn't exist -> create new LFN+SFN entries (as original impl)
1302 char sfn[11];
1303 format_short_name(filename, sfn); // Again, using simple SFN for now
1304 uint8_t checksum = lfn_checksum((uint8_t*)sfn);
1305
1306 uint32_t lfn_count = (kstrlen(filename) + 12) / 13;
1307 uint32_t total_entries = lfn_count + 1;
1308 FAT32_DIR_ENTRY entry_buf[total_entries];
1309
1310 fat32_create_lfn_entries(entry_buf, filename, checksum);
1311
1312 // Create the SFN entry at the end of the buffer
1313 FAT32_DIR_ENTRY* sfn_entry = &entry_buf[lfn_count];
1314 kmemset(sfn_entry, 0, sizeof(FAT32_DIR_ENTRY));
1315 kmemcpy(sfn_entry->name, sfn, 11);
1316 sfn_entry->attr = 0; // Not a directory, archive bit can be set.
1317 // set file size & first_cluster (first_cluster may be zero for size==0)
1318 uint32_t final_size = size;
1319 if (mode == WRITE_MODE_APPEND_EXISTING && exists) final_size = existing_entry.file_size + size;
1320 sfn_entry->file_size = final_size;
1321 sfn_entry->fst_clus_lo = (uint16_t)first_cluster;
1322 sfn_entry->fst_clus_hi = (uint16_t)(first_cluster >> 16);
1323 TIME_ENTRY currTime = get_time();
1324 uint16_t fat32date = fat32_encode_date(currTime.year, currTime.month, currTime.day);
1325 uint16_t fat32time = fat32_encode_time(currTime.hour, currTime.minute, currTime.second);
1326 sfn_entry->wrt_date = fat32date;
1327 sfn_entry->wrt_time = fat32time;
1328 sfn_entry->crt_date = fat32date;
1329 sfn_entry->crt_time = fat32time;
1330 sfn_entry->lst_acc_date = fat32date;
1331
1332 if (exists && located) {
1333 // Update existing on-disk SFN+LFN in-place: overwrite SFN fields + file_size + fst_clus (but preserve other fields).
1334 void* write_buf = MtAllocateVirtualMemory(512, 512);
1335 if (!write_buf) return MT_NO_MEMORY;
1336 status = read_sector(located_sector, write_buf);
1337 if (MT_FAILURE(status)) { MtFreeVirtualMemory(write_buf); return status; }
1338
1339 FAT32_DIR_ENTRY* entries = (FAT32_DIR_ENTRY*)write_buf;
1340 // The located_index is the start (LFN or SFN). We assumed read_lfn returned consumed entries for that start.
1341 // If the located entry was an LFN chain, the SFN is at located_index + (located_consumed - 1)
1342 uint32_t sfn_pos = located_index + (located_consumed ? (located_consumed - 1) : 0);
1343 // Update SFN's cluster and filesize
1344 entries[sfn_pos].fst_clus_lo = (uint16_t)first_cluster;
1345 entries[sfn_pos].fst_clus_hi = (uint16_t)(first_cluster >> 16);
1346 entries[sfn_pos].file_size = final_size;
1347 fat32date = fat32_encode_date(currTime.year, currTime.month, currTime.day);
1348 fat32time = fat32_encode_time(currTime.hour, currTime.minute, currTime.second);
1349 entries[sfn_pos].wrt_date = fat32date;
1350 entries[sfn_pos].wrt_time = fat32time;
1351 entries[sfn_pos].crt_date = fat32date;
1352 entries[sfn_pos].crt_time = fat32time;
1353 entries[sfn_pos].lst_acc_date = fat32date;
1354
1355 // If original file had no LFN and we're writing a long-name (lfn_count>0), we cannot expand in-place safely.
1356 // In that case we fallback to find free slots and write a new LFN+SFN pair and mark old entries deleted.
1357 bool can_update_inplace = (lfn_count == 0) || (located_consumed == total_entries);
1358 if (!can_update_inplace) {
1359 // mark old consumed entries deleted (0xE5), then write new entries in free slots
1360 for (uint32_t k = 0; k < located_consumed; ++k) {
1361 ((uint8_t*)entries[located_index + k].name)[0] = DELETED_DIR_ENTRY;
1362 }
1363 status = write_sector(located_sector, write_buf);
1364 if (MT_FAILURE(status)) { MtFreeVirtualMemory(write_buf); return status; }
1365 MtFreeVirtualMemory(write_buf);
1366
1367 // find new free slots and write new entries (reuse original creation path)
1368 uint32_t entry_sector, entry_index;
1369 if (!fat32_find_free_dir_slots(parent_cluster, total_entries, &entry_sector, &entry_index)) {
1370 // cleanup clusters if we've allocated them
1371 return MT_FAT32_DIR_FULL;
1372 }
1373 void* write_buf2 = MtAllocateVirtualMemory(512, 512);
1374 if (!write_buf2) { return MT_NO_MEMORY; }
1375 status = read_sector(entry_sector, write_buf2);
1376 if (MT_FAILURE(status)) { MtFreeVirtualMemory(write_buf2); return status; }
1377 kmemcpy(&((FAT32_DIR_ENTRY*)write_buf2)[entry_index], entry_buf, total_entries * sizeof(FAT32_DIR_ENTRY));
1378 status = write_sector(entry_sector, write_buf2);
1379 MtFreeVirtualMemory(write_buf2);
1380 return status;
1381 }
1382 else {
1383 // safe to update in-place: write sector
1384 status = write_sector(located_sector, write_buf);
1385 MtFreeVirtualMemory(write_buf);
1386 return status;
1387 }
1388 }
1389 else {
1390 // File did not exist: find space and write the entries (original flow)
1391 uint32_t entry_sector, entry_index;
1392 if (!fat32_find_free_dir_slots(parent_cluster, total_entries, &entry_sector, &entry_index)) {
1393 // If we allocated new clusters earlier, free them (best effort)
1394 if (first_cluster) fat32_free_cluster_chain(first_cluster);
1395 return MT_FAT32_DIR_FULL;
1396 }
1397
1398 void* write_buf = MtAllocateVirtualMemory(512, 512);
1399 if (!write_buf) return MT_NO_MEMORY;
1400 status = read_sector(entry_sector, write_buf);
1401 if (MT_FAILURE(status)) { MtFreeVirtualMemory(write_buf); return status; }
1402 kmemcpy(&((FAT32_DIR_ENTRY*)write_buf)[entry_index], entry_buf, total_entries * sizeof(FAT32_DIR_ENTRY));
1403 status = write_sector(entry_sector, write_buf);
1404
1405 // free write_buf
1406 MtFreeVirtualMemory(write_buf);
1407 return status;
1408 }
1409
1410 // Should not reach here
1411 return MT_GENERAL_FAILURE;
1412}

1414MTSTATUS fat32_list_directory(const char* path, char* listings, size_t max_len) {
1415 tracelast_func("fat32_list_directory");
1416 MTSTATUS status;
1417 // 1. Find the directory entry for the given path to get its starting cluster.
1418 FAT32_DIR_ENTRY dir_entry;
1419 if (!fat32_find_entry(path, &dir_entry, NULL) || !(dir_entry.attr & ATTR_DIRECTORY)) {
1420 gop_printf(0xFFFF0000, "Error: Directory not found or path is not a directory: %s\n", path);
1421 return MT_FAT32_DIRECTORY_NOT_FOUND;
1422 }
1423
1424 uint32_t cluster = (uint32_t)((dir_entry.fst_clus_hi << 16) | dir_entry.fst_clus_lo);
1425 if (cluster == 0) { // Root directory special case on some FAT16/12, but for FAT32 it should be root_cluster.
1426 cluster = fs.root_cluster;
1427 }
1428
1429 void* buf = MtAllocateVirtualMemory(512, 512);
1430 if (!buf) return MT_NO_MEMORY;
1431
1432 if (max_len > 0) listings[0] = '\0';
1433 size_t used = 0;
1434
1435 do {
1436 uint32_t sector = first_sector_of_cluster(cluster);
1437 bool end_of_dir = false;
1438
1439 for (uint32_t i = 0; i < fs.sectors_per_cluster; ++i) {
1440 status = read_sector(sector + i, buf);
1441 if (MT_FAILURE(status)) { MtFreeVirtualMemory(buf); return status; }
1442
1443 FAT32_DIR_ENTRY* dir = (FAT32_DIR_ENTRY*)buf;
1444 uint32_t entries = fs.bytes_per_sector / sizeof(*dir);
1445
1446 for (uint32_t j = 0; j < entries; ) {
1447 FAT32_DIR_ENTRY* current_entry = &dir[j];
1448
1449 if (current_entry->name[0] == END_OF_DIRECTORY) {
1450 end_of_dir = true;
1451 break; // stop scanning entries in this sector -> will break outer loops below
1452 }
1453
1454 if ((uint8_t)current_entry->name[0] == DELETED_DIR_ENTRY ||
1455 (current_entry->name[0] == '.' && (current_entry->name[1] == '\0' || current_entry->name[1] == '.'))) {
1456 j++;
1457 continue;
1458 }
1459
1460 char lfn_name[MAX_LFN_LEN];
1461 uint32_t consumed = 0;
1462 FAT32_DIR_ENTRY* sfn_entry = read_lfn(current_entry, entries - j, lfn_name, &consumed);
1463 char line_buf[256];
1464 if (sfn_entry) {
1465 if (sfn_entry->attr & ATTR_DIRECTORY) {
1466 ksnprintf(line_buf, sizeof(line_buf), "<DIR> %s\n", lfn_name);
1467 }
1468 else {
1469 ksnprintf(line_buf, sizeof(line_buf), "%s (%u bytes)\n", lfn_name, sfn_entry->file_size);
1470 }
1471 // safe append: compute remaining and append up to remaining-1
1472 size_t avail = (used < max_len) ? (max_len - used) : 0;
1473 if (avail > 1) {
1474 // write directly into listings+used
1475 ksnprintf(listings + used, avail, "%s", line_buf);
1476 used = kstrlen(listings);
1477 }
1478 j += consumed;
1479 }
1480 else {
1481 j++;
1482 }
1483 }
1484
1485 if (end_of_dir) break;
1486 }
1487
1488 if (end_of_dir) break;
1489
1490 cluster = fat32_read_fat(cluster);
1491 } while (cluster < FAT32_EOC_MIN);
1492
1493 MtFreeVirtualMemory(buf);
1494 return MT_SUCCESS;
1495}

1497// Check that a directory cluster contains only '.' and '..' (and deleted entries).
1498// Returns true if empty ,false if non-empty or error.
1499bool fat32_directory_is_empty(const char* path) {
1500 tracelast_func("fat32_directory_is_empty");
1501
1502 FAT32_DIR_ENTRY entry;
1503 uint32_t parent_cluster = 0;
1504 fat32_find_entry(path, &entry, &parent_cluster);
1505
1506 uint32_t dir_cluster = get_dir_cluster(&entry);
1507 if (dir_cluster == 0) return false;
1508
1509 void* buf = MtAllocateVirtualMemory(512, 512);
1510 if (!buf) return false;
1511
1512 uint32_t cluster = dir_cluster;
1513 MTSTATUS status;
1514 do {
1515 uint32_t sector_lba = first_sector_of_cluster(cluster);
1516 for (uint32_t s = 0; s < fs.sectors_per_cluster; ++s) {
1517 status = read_sector(sector_lba + s, buf);
1518 if (MT_FAILURE(status)) { MtFreeVirtualMemory(buf); return false; }
1519
1520 FAT32_DIR_ENTRY* entries = (FAT32_DIR_ENTRY*)buf;
1521 uint32_t entries_per_sector = fs.bytes_per_sector / sizeof(FAT32_DIR_ENTRY);
1522
1523 for (uint32_t j = 0; j < entries_per_sector; ) {
1524 uint8_t first = (uint8_t)entries[j].name[0];
1525
1526 if (first == END_OF_DIRECTORY) { MtFreeVirtualMemory(buf); return true; } // no more entries
1527 if (first == DELETED_DIR_ENTRY) { j++; continue; }
1528
1529 // Build full name (LFN or SFN)
1530 char lfn_buf[MAX_LFN_LEN];

1533 if (!sfn) { j++; continue; }
1534
1535 // skip '.' and '..'
1536 if ((unsigned char)sfn->name[0] == '.') {
1537 j += consumed;
1538 continue;
1539 }
1540
1541 // There is a non-deleted entry that is not '.'/'..' -> directory not empty
1542 MtFreeVirtualMemory(buf);
1543 return false;
1544 }
1545 }
1546 cluster = fat32_read_fat(cluster);
1547 } while (cluster < FAT32_EOC_MIN);
1548
1549 MtFreeVirtualMemory(buf);
1550 return true;
1551}

1553// Mark the SFN and all preceding LFN entries for `filename` in parent_cluster as deleted.
1554// `path` is the full path. parent_cluster is cluster of parent directory.
1555// Returns true on success (sector written), false otherwise.
1556static bool mark_entry_and_lfns_deleted(const char* path, uint32_t parent_cluster) {
1557 tracelast_func("mark_entry_and_lfns_deleted");
1558 // extract filename (last component)
1559 char path_copy[260];
1560 kstrcpy(path_copy, path);
1561 int len = (int)kstrlen(path_copy);
1562 // strip trailing slashes
1563 while (len > 1 && path_copy[len - 1] == '/') { path_copy[--len] = '\0'; }
1564
1565 int last_slash = -1;
1566 for (int i = len - 1; i >= 0; --i) {
1567 if (path_copy[i] == '/') { last_slash = i; break; }
1568 }
1569
1570 const char* filename = (last_slash == -1) ? path_copy : &path_copy[last_slash + 1];
1571
1572 // Prepare SFN format for short-name comparison
1573 char sfn_formatted[11];
1574 format_short_name(filename, sfn_formatted);
1575
1576 void* buf = MtAllocateVirtualMemory(512, 512);
1577 if (!buf) return false;
1578
1579 uint32_t cluster = parent_cluster;
1580 MTSTATUS status;
1581 do {
1582 uint32_t sector_lba = first_sector_of_cluster(cluster);
1583 for (uint32_t s = 0; s < fs.sectors_per_cluster; ++s) {
1584 status = read_sector(sector_lba + s, buf);
1585 if (MT_FAILURE(status)) { MtFreeVirtualMemory(buf); return false; }
1586
1587 FAT32_DIR_ENTRY* entries = (FAT32_DIR_ENTRY*)buf;
1588 uint32_t entries_per_sector = fs.bytes_per_sector / sizeof(FAT32_DIR_ENTRY);
1589
1590 for (uint32_t j = 0; j < entries_per_sector; ) {
1591 uint8_t first = (uint8_t)entries[j].name[0];
1592
1593 if (first == END_OF_DIRECTORY) { MtFreeVirtualMemory(buf); return false; } // not found in parent
1594 if (first == DELETED_DIR_ENTRY) { j++; continue; }
1595
1596 char lfn_buf[MAX_LFN_LEN];
1597 uint32_t consumed = 0;
1598 FAT32_DIR_ENTRY* sfn = read_lfn(&entries[j], entries_per_sector - j, lfn_buf, &consumed);
1599
1600 if (sfn) {
1601 // Match by LFN (exact), LFN (case-insensitive), or SFN bytes
1602 bool match = false;
1603
1604 // 1) exact LFN match
1605 if (kstrcmp(lfn_buf, filename) == 0) {
1606 match = true;
1607 }
1608
1609 // 2) case-insensitive LFN match
1610 if (!match && ci_equal(lfn_buf, filename)) {
1611 match = true;
1612 }
1613
1614 // 3) SFN byte-wise compare (token formatted)
1615 if (!match && cmp_short_name(sfn->name, sfn_formatted)) {
1616 match = true;
1617 }
1618
1619 if (match) {
1620 // Mark all consumed entries (LFN...SFN) as deleted (0xE5)
1621 for (uint32_t k = 0; k < consumed; ++k) {
1622 ((uint8_t*)entries[j + k].name)[0] = DELETED_DIR_ENTRY;
1623 }
1624
1625 // Write sector back to disk
1626 bool ok = write_sector(sector_lba + s, buf);
1627 MtFreeVirtualMemory(buf);
1628 return ok;
1629 }
1630
1631 j += consumed;
1632 continue;
1633 }
1634 else {
1635 // read_lfn failed (corrupted LFN chain?), skip this entry
1636 j++;
1637 }
1638 }
1639 }
1640 cluster = fat32_read_fat(cluster);
1641 } while (cluster < FAT32_EOC_MIN);
1642
1643 MtFreeVirtualMemory(buf);
1644 return false; // not found
1645}

1648// Recursively delete directory contents and free the directory's cluster chain.
1649// This function deletes all children (files & subdirs) found inside dir_cluster,
1650// marks their directory entries as DELETED on disk, and finally frees dir_cluster itself.
1651// Returns true on success, false on any error.
1652static bool fat32_rm_rf_dir(uint32_t dir_cluster) {
1653 tracelast_func("fat32_rm_rf_dir");
1654
1655 if (dir_cluster == 0 || dir_cluster == fs.root_cluster) return false; // never delete root here
1656
1657 void* buf = MtAllocateVirtualMemory(512, 512);
1658 if (!buf) return false;
1659
1660 uint32_t cluster = dir_cluster;
1661 // Iterate cluster chain
1662 MTSTATUS status;
1663 while (cluster < FAT32_EOC_MIN) {
1664 uint32_t sector_lba = first_sector_of_cluster(cluster);
1665
1666 for (uint32_t s = 0; s < fs.sectors_per_cluster; ++s) {
1667 status = read_sector(sector_lba + s, buf);
1668 if (MT_FAILURE(status)) { MtFreeVirtualMemory(buf); return false; }
1669
1670 FAT32_DIR_ENTRY* entries = (FAT32_DIR_ENTRY*)buf;
1671 uint32_t entries_per_sector = fs.bytes_per_sector / sizeof(FAT32_DIR_ENTRY);
1672
1673 for (uint32_t j = 0; j < entries_per_sector; ) {
1674 uint8_t first = (uint8_t)entries[j].name[0];
1675
1676 // End of directory: nothing after this in this directory
1677 if (first == END_OF_DIRECTORY) {
1678 // we can stop scanning this directory entirely
1679 // free buffer and break out to free cluster chain
1680 MtFreeVirtualMemory(buf);
1681 goto free_and_return;
1682 }
1683
1684 // Deleted entry: skip
1685 if (first == DELETED_DIR_ENTRY) { j++; continue; }
1686
1687 // Attempt to read LFN + SFN at this position
1688 char lfn_name[MAX_LFN_LEN];
1689 uint32_t consumed = 0;
1690 FAT32_DIR_ENTRY* sfn = read_lfn(&entries[j], entries_per_sector - j, lfn_name, &consumed);
1691
1692 if (!sfn) {
1693 // corrupted chain or not an entry we can parse: skip single entry
1694 j++;
1695 continue;
1696 }
1697
1698 // Skip '.' and '..' entries
1699 if ((unsigned char)sfn->name[0] == '.') {
1700 j += consumed;
1701 continue;
1702 }
1703
1704 // If directory -> recurse
1705 if (sfn->attr & ATTR_DIRECTORY) {
1706 uint32_t child_cluster = get_dir_cluster(sfn);
1707 if (child_cluster != 0 && child_cluster != 1 && child_cluster != dir_cluster) {
1708 // Recursively delete child directory contents and free its clusters.
1709 if (!fat32_rm_rf_dir(child_cluster)) {
1710 // recursion failed return false
1711 MtFreeVirtualMemory(buf);
1712 return false;
1713 }
1714 // At this point child's clusters are freed by the recursive call.
1715 }
1716 // After child deleted, mark child's LFN+SFN entries as deleted in this parent sector
1717 for (uint32_t k = 0; k < consumed; ++k) {
1718 ((uint8_t*)entries[j + k].name)[0] = DELETED_DIR_ENTRY;
1719 }
1720 // write this sector back
1721 status = write_sector(sector_lba + s, buf);
1722 if (MT_FAILURE(status)) { MtFreeVirtualMemory(buf); return false; }
1723 // advance past consumed entries
1724 j += consumed;
1725 continue;
1726 }
1727 else {
1728 // It's a file: free its cluster chain (if any) then mark entries deleted
1729 uint32_t file_cluster = get_dir_cluster(sfn);
1730 if (file_cluster >= 2) {
1731 if (!fat32_free_cluster_chain(file_cluster)) {
1732 MtFreeVirtualMemory(buf);
1733 return false;
1734 }
1735 }
1736 // mark the LFN+SFN entries as deleted
1737 for (uint32_t k = 0; k < consumed; ++k) {
1738 ((uint8_t*)entries[j + k].name)[0] = DELETED_DIR_ENTRY;
1739 }
1740 // write sector back
1741 status = write_sector(sector_lba + s, buf);
1742 if (MT_FAILURE(status)) { MtFreeVirtualMemory(buf); return false; }
1743 j += consumed;
1744 continue;
1745 }
1746 } // for each entry in sector
1747 } // for each sector in cluster
1748
1749 cluster = fat32_read_fat(cluster);
1750 } // while cluster chain
1751
1752free_and_return:
1753 // Free this directory's own cluster chain (we deleted contents)
1754 if (!fat32_free_cluster_chain(dir_cluster)) {
1755 // if freeing fails, we still consider it an error
1756 return false;
1757 }
1758 return true;
1759}

1761MTSTATUS fat32_delete_directory(const char* path) {
1762 tracelast_func("fat32_delete_directory");
1763
1764 // Find entry & its parent cluster
1765 FAT32_DIR_ENTRY entry;
1766 uint32_t parent_cluster;
1767 if (!fat32_find_entry(path, &entry, &parent_cluster)) return MT_FAT32_DIRECTORY_NOT_FOUND;
1768
1769 // Must be a directory
1770 if (!(entry.attr & ATTR_DIRECTORY)) return MT_FAT32_INVALID_FILENAME;
1771
1772 uint32_t dir_cluster = get_dir_cluster(&entry);

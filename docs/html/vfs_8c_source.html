41MTSTATUS vfs_init(void) {
42 // First initialize other FS Related stuff (FAT32, AHCI, etc..)
43 MTSTATUS status = ahci_init();
44 if (MT_FAILURE(status)) return status;
45
46 // Mount FAT32 on MAIN_FS_DEVICE
47 status = fat32_driver.init(MAIN_FS_DEVICE);
48 if (MT_FAILURE(status)) return status;
49
50 mounted_fs[mount_count++] = (MOUNTED_FS){ .driver = &fat32_driver, .device_id = MAIN_FS_DEVICE, .mount_point = "/" };
51
52 return MT_SUCCESS;
53}

55static MOUNTED_FS* vfs_find_fs_for_path(const char* path) {
56 if (!path) return NULL;
57 for (uint8_t i = 0; i < mount_count; i++) {
58 const char* mount = mounted_fs[i].mount_point;
59 // root mount should match anything
60 if (mount[0] == '/' && mount[1] == '\0') return &mounted_fs[i];
61
62 // compute mount_len
63 size_t mount_len = 0;
64 while (mount[mount_len]) mount_len++;
65
66 // path must be at least mount_len
67 size_t path_len = 0;
68 while (path[path_len]) path_len++;
69 if (path_len < mount_len) continue;
70
71 bool match = true;
72 for (size_t j = 0; j < mount_len; j++) {
73 if (path[j] != mount[j]) { match = false; break; }
74 }
75 if (match) return &mounted_fs[i];
76 }
77 return NULL;
78}

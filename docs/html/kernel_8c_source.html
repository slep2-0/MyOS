35void copy_memory_map(BOOT_INFO* boot_info) {
36 if (!boot_info || !boot_info->MemoryMap) return;
37 if (boot_info->MapSize > MAX_MEMORY_MAP_SIZE) {
38 // handle error, memory map too big
39 MtBugcheck(NULL, NULL, MEMORY_MAP_SIZE_OVERRUN, 0, false);
40 }
41
42 // Copy the entire memory map into the static buffer
43 kmemcpy(memory_map_copy, boot_info->MemoryMap, boot_info->MapSize);
44
45 boot_info_local.MemoryMap = memory_map_copy;
46 boot_info_local.MapSize = boot_info->MapSize;
47 boot_info_local.DescriptorSize = boot_info->DescriptorSize;
48 boot_info_local.DescriptorVersion = boot_info->DescriptorVersion;
49}

51void copy_gop(BOOT_INFO* boot_info) {
52 if (!boot_info || !boot_info->Gop) return;
53
54 // Copy the GOP data to a local global variable
55 gop_local = *(boot_info->Gop);
56
57 // Update all relevant pointers to point to the local copy
58 boot_info->Gop = &gop_local;
59 boot_info_local.Gop = &gop_local;

63void init_boot_info(BOOT_INFO* boot_info) {
64 if (!boot_info) return;
65
66 copy_memory_map(boot_info);
67 copy_gop(boot_info);
68 if (boot_info->AhciCount > MAX_AHCI_CONTROLLERS) {
69 MtBugcheck(NULL, NULL, BAD_AHCI_COUNT, 0, false);
70 }
71 for (uint32_t i = 0; i < boot_info->AhciCount; i++) {
72 ahci_bases_local[i] = boot_info->AhciBarBases[i];

74 boot_info_local.AhciBarBases = ahci_bases_local;
75 boot_info_local.AhciCount = boot_info->AhciCount;
76 boot_info_local.KernelStackTop = boot_info->KernelStackTop;
77 boot_info_local.Pml4Phys = boot_info->Pml4Phys;
78}

80static void InitialiseControlRegisters(void) {
81
82 /* CR0 */
83 unsigned long cr0 = __read_cr0();
84 cr0 |= (1UL << 16); // Set bit 16 (WRITE PROTECT), so when the kernel touches read only memory it would #PF.
85#ifdef DISABLE_CACHE
86 cr0 |= (1UL << 30); // Set bit 30 (CACHE DISABLE).
87#endif
88 __write_cr0(cr0);
89
90 /* CR4 */
91 unsigned long cr4 = __read_cr4();
92 cr4 |= (1UL << 11); // Set bit 11 - User Mode Instruction Prevention. This'll be useful against user mode attacks to locate IDT/GDT/LDT...
93 __write_cr4(cr4);
94}

96void InitCPU(void) {
97 cpu.currentIrql = PASSIVE_LEVEL;
98 cpu.schedulerEnabled = NULL; // since NULL is 0, it would be false.
99 cpu.currentThread = NULL;
100 cpu.readyQueue.head = cpu.readyQueue.tail = NULL;
101 __writemsr(IA32_KERNEL_GS_BASE, (uint64_t) & cpu);
102 spinlock_init(&cpu.readyQueue.lock);
103}

105static inline bool interrupts_enabled(void) {
106 unsigned long flags;
107 __asm__ __volatile__("pushfq; popq %0"

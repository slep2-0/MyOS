23// Interrupt Request Level - Changing this will mask IRQ lines, as well as disabling features of the OS/CPU, like disabling the scheduler when raisng to >= DISPATCH_LEVEL.
24typedef enum _IRQL {
25 PASSIVE_LEVEL = 0, // Normal thread execution, all interrupts enabled.
26 DISPATCH_LEVEL = 2, // Scheduler disabled, page faults fatal. (page faults are always fatal for now, until I implement an exception handler TODO)
27 // Device DIRQLs (chosen so IRQn + DIRQL = PROFILE_LEVEL (27))
28 DIRQL_SECONDARY_ATA = 12, // IRQ15 – Secondary ATA Channel
29 DIRQL_PRIMARY_ATA = 13, // IRQ14 – Primary ATA Channel
30 DIRQL_FPU = 14, // IRQ13 – FPU / Coprocessor
31 DIRQL_MOUSE = 15, // IRQ12 – Mouse
32 DIRQL_PERIPHERAL11 = 16, // IRQ11 – Free for peripherals
33 DIRQL_PERIPHERAL10 = 17, // IRQ10 – Free for peripherals
34 DIRQL_PERIPHERAL9 = 18, // IRQ9 – Free / redirected cascade
35 DIRQL_RTC = 19, // IRQ8 – RTC / CMOS Alarm
36 DIRQL_LPT1 = 20, // IRQ7 – LPT1 / Printer
37 DIRQL_FLOPPY = 21, // IRQ6 – Floppy Disk
38 DIRQL_SOUND_LPT2 = 22, // IRQ5 – Sound Card / LPT2
39 DIRQL_COM1 = 23, // IRQ4 – Serial COM1
40 DIRQL_COM2 = 24, // IRQ3 – Serial COM2
41 DIRQL_CASCADE = 25, // IRQ2 – Cascade (IRQs 8–15)
42 DIRQL_KEYBOARD = 26, // IRQ1 – Keyboard
43 DIRQL_TIMER = 27, // IRQ0 – System Timer
44 PROFILE_LEVEL = 27, // Profile timer interrupts (alias of DIRQL_TIMER)
45 CLOCK_LEVEL = 28, // Clock/timer interrupts (second-level timer) (actual clock IRQ timer, for scheduler, time-of-day clock, and general timers, even context switching)
46 SYNCH_LEVEL = 29, // Synchronization level (internal kernel use) (unused in my kernel (until SMP), this level is used for multi core CPU synchronization)
47 POWER_LEVEL = 30, // Power failure interrupts
48 HIGH_LEVEL = 31 // NMI and machine-check (non-maskable) (also gets set at bugchecks, to mask all interrupts)
49} IRQL;

131// --- Timer calibration and init ---
132// NOTE: the APIC timer is a downward counter. Strategy:
133// 1. Set divide to known divisor.
134// 2. Write initcount = 0xFFFFFFFF.
135// 3. Wait EXACTLY 100 ms via PIT/HPET.
136// 4. curr = read current count -> ticks_in_100ms = start - curr
137// 5. ticks_per_period(10ms) = ticks_in_100ms / 10
138// 6. Program LVT timer to periodic and initial count = ticks_per_period
139//
140#define APIC_LVT_TIMER_PERIODIC (1U << 17)
141#define APIC_TIMER_MASKED (1U << 16)
142
143static uint32_t calibrate_lapic_ticks_per_10ms(void) {
144 // choose divide config: here set encode 0x3 (divide by 16). Adjust if needed.
145 lapic_mmio_write(LAPIC_TIMER_DIV, 0x3);
146
147 const uint32_t start = 0xFFFFFFFFU;
148 lapic_mmio_write(LAPIC_TIMER_INITCNT, start);
149
150 pit_sleep_ms(100);
151
152 uint32_t curr = lapic_mmio_read(LAPIC_TIMER_CURRCNT);
153 uint32_t ticks = start - curr;
154 if (ticks == 0) return 0;
155 return ticks / 10; // ticks per 10ms -> for 100Hz (10ms period)
156}

158int init_lapic_timer(uint32_t hz) {
159 if (hz == 0) return -1;
160 map_lapic();
161
162 // calibrate using 100ms window
163 uint32_t ticks_per_10ms = calibrate_lapic_ticks_per_10ms();
164 if (ticks_per_10ms == 0) return -2;
165
166 // compute target initial count
167 // desired_period_ms = 1000 / hz
168 uint32_t period_ms = 1000 / hz;
169 // ticks_per_10ms: ticks per 10ms, so ticks_per_ms = ticks_per_10ms / 10
170 // initial_count = ticks_per_ms * period_ms = ticks_per_10ms * period_ms / 10
171 uint64_t initial = ((uint64_t)ticks_per_10ms * (uint64_t)period_ms) / 10ULL;
172 if (initial == 0) initial = 1;
173
174 // mask the timer while programming
175 lapic_mmio_write(LAPIC_LVT_TIMER, APIC_LVT_TIMER_PERIODIC | 0xEF /* IDT vector 0xEF */);
176 lapic_mmio_write(LAPIC_TIMER_INITCNT, (uint32_t)initial);
177 return 0;
178}

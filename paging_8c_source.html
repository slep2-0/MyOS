23static uint64_t* allocate_page_table(void) {
24 tracelast_func("allocate_page_table");
25 // CHECK IRQL.
26 enforce_max_irql(PASSIVE_LEVEL);
27 // 1) if we still have one of the linker‑reserved tables, carve that out:
28 if (next_pt + PAGE_SIZE_4K <= end_pt) {
29 uint64_t* table = (uint64_t*)next_pt;
30 next_pt += PAGE_SIZE_4K;
31 kmemset(table, 0, PAGE_SIZE_4K);
32 return table;
33 }
34
35 // 2) otherwise fall back on the frame‐bitmap
36 void* phys = alloc_frame();
37 if (!phys) {
38 bugcheck_system(NULL, NULL, BAD_PAGING, 0, false);
39 return NULL;
40 }
41 // zero it before use
42 kmemset(phys, 0, PAGE_SIZE_4K);
43 return (uint64_t*)phys;
44}

52void map_range_identity(uint64_t start, uint64_t end, uint64_t flags) {
53 tracelast_func("map_range_identity");
54 enforce_max_irql(PASSIVE_LEVEL);
55 for (uint64_t addr = start; addr < end; addr += PAGE_SIZE_4K) {
56 map_page((void*)addr, (void*)addr, flags);
57 }

81void paging_init(void) {
82 tracelast_func("paging_init");
83 enforce_max_irql(PASSIVE_LEVEL);
84 // zero your PML4…
85 kmemset(pml4, 0, PAGE_SIZE_4K);
86 // carve out the first few tables (PML4→PDPT→PD→PT)
87 uint64_t* pdpt = allocate_page_table();
88 uint64_t* pd = allocate_page_table();
89 uint64_t* pt = allocate_page_table();
90 pml4[0] = (uint64_t)pdpt | PAGE_PRESENT | PAGE_RW;
91 pdpt[0] = (uint64_t)pd | PAGE_PRESENT | PAGE_RW;
92 pd[0] = (uint64_t)pt | PAGE_PRESENT | PAGE_RW;
93
94 const uint64_t flags = PAGE_PRESENT | PAGE_RW;
95
96 map_range_identity((uintptr_t)&__pml4_start,
97 (uintptr_t)&__pml4_start + PAGE_SIZE_4K,
98 flags);
99
100 map_range_identity((uintptr_t)&__pt_start,
101 (uintptr_t)&__pt_end,
102 flags);
103
104 map_range_identity((uintptr_t)&kernel_start,
105 (uintptr_t)&kernel_end + 1,
106 flags);
107
108 extern uint8_t __stack_start[], __stack_end[];
109 map_range_identity((uintptr_t)__stack_start,
110 (uintptr_t)__stack_end,
111 flags);
112
113 {
114 uint64_t fb = gop_local.FrameBufferBase;
115 uint64_t end = fb + gop_local.FrameBufferSize;
116 map_range_identity(fb, end, flags);
117 }
118
119 extern IDT_ENTRY64 IDT[];
120 map_range_identity((uintptr_t)IDT,
121 (uintptr_t)IDT + sizeof(IDT_ENTRY64) * IDT_ENTRIES,
122 flags);
123
124 // …then load CR3 and enable paging as before…
125 enable_paging((uint64_t)pml4);
126}

129// Map a 4KiB page: map virtual address to physical with given flags (PAGE_RW, PAGE_USER, etc)
130void map_page(void* virtualaddress, void* physicaladdress, uint64_t flags) {
131 tracelast_func("map_page");
132 enforce_max_irql(PASSIVE_LEVEL);
133 uint64_t va = (uint64_t)virtualaddress;
134 uint64_t pa = (uint64_t)physicaladdress;
135
136 size_t pml4_i = get_pml4_index(va);
137 size_t pdpt_i = get_pdpt_index(va);
138 size_t pd_i = get_pd_index(va);
139 size_t pt_i = get_pt_index(va);
140
141 // Traverse or allocate PDPT
142 uint64_t* pdpt;
143 if (!(pml4[pml4_i] & PAGE_PRESENT)) {
144 pdpt = allocate_page_table();
145 if (!pdpt) return; // fail
146 pml4[pml4_i] = ((uint64_t)pdpt) | PAGE_PRESENT | PAGE_RW | PAGE_USER;
147 }
148 else {
149 pdpt = (uint64_t*)(pml4[pml4_i] & ~0xFFFULL);
150 }
151
152 // Traverse or allocate PD
153 uint64_t* pd;
154 if (!(pdpt[pdpt_i] & PAGE_PRESENT)) {
155 pd = allocate_page_table();
156 if (!pd) return; // fail
157 pdpt[pdpt_i] = ((uint64_t)pd) | PAGE_PRESENT | PAGE_RW | PAGE_USER;
158 }
159 else {
160 pd = (uint64_t*)(pdpt[pdpt_i] & ~0xFFFULL);
161 }
162
163 // Traverse or allocate PT
164 uint64_t* pt;
165 if (!(pd[pd_i] & PAGE_PRESENT)) {
166 pt = allocate_page_table();
167 if (!pt) return; // fail
168 pd[pd_i] = ((uint64_t)pt) | PAGE_PRESENT | PAGE_RW | PAGE_USER;
169 }
170 else {
171 pt = (uint64_t*)(pd[pd_i] & ~0xFFFULL);
172 }
173
174 // Map the page
175 pt[pt_i] = (pa & ~0xFFFULL) | (flags & (PAGE_RW | PAGE_USER)) | PAGE_PRESENT;
176
177 // Flush TLB for this virtual address
178 // only flush if paging is already on:
179 uint64_t cr0;
180 __asm__ volatile("mov %%cr0, %0" : "=r"(cr0));
181 if (cr0 & 0x80000000) {
182 invlpg((void*)va);
183 }
184}

186// Unmap a page (remove mapping and free frame)
187bool unmap_page(void* virtualaddress) {
188 tracelast_func("unmap_page");
189 enforce_max_irql(PASSIVE_LEVEL);
190 uint64_t va = (uint64_t)virtualaddress;
191
192 size_t pml4_i = get_pml4_index(va);
193 size_t pdpt_i = get_pdpt_index(va);
194 size_t pd_i = get_pd_index(va);
195 size_t pt_i = get_pt_index(va);
196
197 if (!(pml4[pml4_i] & PAGE_PRESENT)) return false;
198 uint64_t* pdpt = (uint64_t*)(pml4[pml4_i] & ~0xFFFULL);
199
200 if (!(pdpt[pdpt_i] & PAGE_PRESENT)) return false;
201 uint64_t* pd = (uint64_t*)(pdpt[pdpt_i] & ~0xFFFULL);
202
203 if (!(pd[pd_i] & PAGE_PRESENT)) return false;
204 uint64_t* pt = (uint64_t*)(pd[pd_i] & ~0xFFFULL);
205
206 if (!(pt[pt_i] & PAGE_PRESENT)) return false;
207
208 void* phys_addr = (void*)(pt[pt_i] & ~0xFFFULL);
209
210 // Clear the entry
211 pt[pt_i] = 0;
212
213 // only flush if paging is already on:
214 uint64_t cr0;
215 __asm__ volatile("mov %%cr0, %0" : "=r"(cr0));
216 if (cr0 & 0x80000000) {
217 invlpg((void*)va);
218 }
219
220 // Free physical frame if applicable
221 free_frame(phys_addr);
222
223 return true;
224}

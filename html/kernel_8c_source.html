33void copy_memory_map(BOOT_INFO* boot_info) {
34 if (!boot_info || !boot_info->MemoryMap) return;
35 if (boot_info->MapSize > MAX_MEMORY_MAP_SIZE) {
36 // handle error, memory map too big
37 MtBugcheck(NULL, NULL, MEMORY_MAP_SIZE_OVERRUN, 0, false);
38 }
39
40 // Copy the entire memory map into the static buffer
41 kmemcpy(memory_map_copy, boot_info->MemoryMap, boot_info->MapSize);
42
43 boot_info_local.MemoryMap = memory_map_copy;
44 boot_info_local.MapSize = boot_info->MapSize;
45 boot_info_local.DescriptorSize = boot_info->DescriptorSize;
46 boot_info_local.DescriptorVersion = boot_info->DescriptorVersion;
47}

49void copy_gop(BOOT_INFO* boot_info) {

61void init_boot_info(BOOT_INFO* boot_info) {
62 if (!boot_info) return;
63
64 copy_memory_map(boot_info);
65 copy_gop(boot_info);
66 if (boot_info->AhciCount > MAX_AHCI_CONTROLLERS) {
67 MtBugcheck(NULL, NULL, BAD_AHCI_COUNT, 0, false);

70 ahci_bases_local[i] = boot_info->AhciBarBases[i];

73 boot_info_local.AhciCount = boot_info->AhciCount;
74 boot_info_local.KernelStackTop = boot_info->KernelStackTop;
75 boot_info_local.Pml4Phys = boot_info->Pml4Phys;

78static void InitialiseControlRegisters(void) {
79
80 /* CR0 */
81 unsigned long cr0 = __read_cr0();
82 cr0 |= (1UL << 16); // Set bit 16 (WRITE PROTECT), so when the kernel touches read only memory it would #PF.
83#ifdef DEBUG
84 cr0 |= (1UL << 30); // Set bit 30 (CACHE DISABLE).
85#endif
86 __write_cr0(cr0);
87
88 /* CR4 */
89 unsigned long cr4 = __read_cr4();
90 cr4 |= (1UL << 11); // Set bit 11 - User Mode Instruction Prevention. This'll be useful against user mode attacks to locate IDT/GDT/LDT...
91 __write_cr4(cr4);
92}

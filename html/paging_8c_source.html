24static inline uint64_t canonical_high(uint64_t addr) {
25 // If bit 47 is set, set all higher bits
26 if (addr & (1ULL << 47)) {
27 return addr | 0xFFFF000000000000ULL;
28 }
29 return addr;
30}

34static inline uint64_t* pml4_from_recursive(void) {
35 uint64_t va = ((uint64_t)RECURSIVE_INDEX << 39) |
36 ((uint64_t)RECURSIVE_INDEX << 30) |
37 ((uint64_t)RECURSIVE_INDEX << 21) |

96// Corrected map_page function
97void map_page(void* virtualaddress, uintptr_t physicaladdress, uint64_t flags) {
98 tracelast_func("map_page");
99 uint64_t va = (uint64_t)virtualaddress;
100 uintptr_t pa = physicaladdress;
101 BUGCHECK_ADDITIONALS addt = { 0 };
102 size_t pml4_i = get_pml4_index(va);
103 size_t pdpt_i = get_pdpt_index(va);
104 size_t pd_i = get_pd_index(va);
105 size_t pt_i = get_pt_index(va);
106
107 uint64_t* pml4_va = pml4_from_recursive();
108
109 // 2. Ensure PDPT exists
110 if (!(pml4_va[pml4_i] & PAGE_PRESENT)) {
111 ksnprintf(addt.str, sizeof(addt.str), "In PML4, VA: %p, PA: %p, FLAGS: %d", virtualaddress, physicaladdress, flags);
112 MtBugcheckEx(NULL, NULL, BAD_PAGING, &addt, true);
113 }
114 uint64_t* pdpt_va = pdpt_from_recursive(pml4_i);
115
116 // 3. Ensure PD exists
117 if (!(pdpt_va[pdpt_i] & PAGE_PRESENT)) {
118 ksnprintf(addt.str, sizeof(addt.str), "In PDPT, VA: %p, PA: %p, FLAGS: %d", virtualaddress, physicaladdress, flags);
119 MtBugcheckEx(NULL, NULL, BAD_PAGING, &addt, true);
120 }
121 uint64_t* pd_va = pd_from_recursive(pml4_i, pdpt_i);
122
123 // 4. Ensure PT exists
124 if (!(pd_va[pd_i] & PAGE_PRESENT)) {
125 ksnprintf(addt.str, sizeof(addt.str), "In PD, VA: %p, PA: %p, FLAGS: %d", virtualaddress, physicaladdress, flags);
126 MtBugcheckEx(NULL, NULL, BAD_PAGING, &addt, true);
127 }
128 uint64_t* pt_va = pt_from_recursive(pml4_i, pdpt_i, pd_i);
129
130 // 5. Map the page
131 pt_va[pt_i] = (pa & ~0xFFFULL) | flags;
132
133 // 6. Flush the TLB for the target virtual address
134 invlpg(virtualaddress);
135}

138bool unmap_page(void* virtualaddress) {
139 tracelast_func("unmap_page");
140 uint64_t rip;
141 GET_RIP(rip);
142 enforce_max_irql(DISPATCH_LEVEL, (void*)rip);
143 uint64_t va = (uint64_t)virtualaddress;
144
145 size_t pml4_i = get_pml4_index(va);
146 size_t pdpt_i = get_pdpt_index(va);
147 size_t pd_i = get_pd_index(va);
148 size_t pt_i = get_pt_index(va);
149
150 // FIX: Use recursive helpers to get VIRTUAL addresses of tables
151 uint64_t* pml4 = pml4_from_recursive();
152 if (!(pml4[pml4_i] & PAGE_PRESENT)) return false;
153
154 uint64_t* pdpt = pdpt_from_recursive(pml4_i);
155 if (!(pdpt[pdpt_i] & PAGE_PRESENT)) return false;
156
157 uint64_t* pd = pd_from_recursive(pml4_i, pdpt_i);
158 if (!(pd[pd_i] & PAGE_PRESENT)) return false;
159
160 uint64_t* pt = pt_from_recursive(pml4_i, pdpt_i, pd_i);
161 if (!(pt[pt_i] & PAGE_PRESENT)) return false;
162
163 uintptr_t phys_addr = (uintptr_t)(pt[pt_i] & ~0xFFFULL);
164
165 free_frame(phys_addr); // Free the underlying physical frame
166 pt[pt_i] = 0; // Clear the entry
167 invlpg(virtualaddress); // Invalidate the address
168 return true;
169}

314uintptr_t MtTranslateVirtualToPhysical(void* virtualaddress) {
315 uint64_t va = (uint64_t)virtualaddress;
316 size_t pml4_i = get_pml4_index(va), pdpt_i = get_pdpt_index(va);
317 size_t pd_i = get_pd_index(va), pt_i = get_pt_index(va);
318 size_t offset = get_offset(va);
319
320 uint64_t* pml4 = pml4_from_recursive();
321 if (!(pml4[pml4_i] & PAGE_PRESENT)) return 0;
322
323 uint64_t* pdpt = pdpt_from_recursive(pml4_i);
324 uint64_t pdpt_entry = pdpt[pdpt_i];
325 if (!(pdpt_entry & PAGE_PRESENT)) return 0;
326 // 1GiB page?
327 if (pdpt_entry & PAGE_PS) {
328 uintptr_t base = (pdpt_entry & ~((1ULL << 30) - 1)); // mask lower 30 bits
329 return base + (va & ((1ULL << 30) - 1));
330 }
331
332 uint64_t* pd = pd_from_recursive(pml4_i, pdpt_i);
333 uint64_t pd_entry = pd[pd_i];
334 if (!(pd_entry & PAGE_PRESENT)) return 0;
335 // 2MiB page?
336 if (pd_entry & PAGE_PS) {
337 uintptr_t base = pd_entry & ~((1ULL << 21) - 1);
338 return base + (va & ((1ULL << 21) - 1));
339 }
340
341 uint64_t* pt = pt_from_recursive(pml4_i, pdpt_i, pd_i);
342 uint64_t pt_entry = pt[pt_i];
343 if (!(pt_entry & PAGE_PRESENT)) return 0;
344
345 uintptr_t base = pt_entry & ~0xFFFULL;
346 return base + offset;
347}

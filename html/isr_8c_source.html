22void isr_handler64(int vec_num, CTX_FRAME* ctx, INT_FRAME* intfr) {
23 char buf[256];
24 ksnprintf(buf, sizeof(buf), "INTERRUPT: %d", vec_num);
25 tracelast_func(buf);
26 IRQL oldIrql;
27
28 // Save if the scheduler was enabled or not before raising to >= DISPATCH_LEVEL (because in dispatch_level the scheduler gets disabled to disable pre-emption)
29 bool schedulerEnabled = cpu.schedulerEnabled;
30
31 ctx->rip = intfr->rip;
32 ctx->rsp = intfr->rsp;
33 intfr->vector = vec_num;
34
35 switch (vec_num) {
36 case EXCEPTION_DIVIDE_BY_ZERO:
37 dividebyzero_handler(ctx, intfr);
38 break;
39 case EXCEPTION_SINGLE_STEP:
40 debugsinglestep_handler(ctx, intfr);
41 break;
42 case EXCEPTION_NON_MASKABLE_INTERRUPT:
43 _MtSetIRQL(HIGH_LEVEL); // Non Maskable Interrupt - basically when the CPU encounters a hardware fault, cannot be masked, very alarming.
44 nmi_handler(ctx, intfr);
45 break;
46 case EXCEPTION_BREAKPOINT:
47 breakpoint_handler(ctx, intfr);
48 break;
49 case EXCEPTION_OVERFLOW:
50 overflow_handler(ctx, intfr);
51 break;
52 case EXCEPTION_BOUNDS_CHECK:
53 boundscheck_handler(ctx, intfr);
54 break;
55 case EXCEPTION_INVALID_OPCODE:
56 invalidopcode_handler(ctx, intfr);
57 break;
58 case EXCEPTION_NO_COPROCESSOR:
59 nocoprocessor_handler(ctx, intfr);
60 break;
61 case EXCEPTION_DOUBLE_FAULT:
62 _MtSetIRQL(HIGH_LEVEL);
63 doublefault_handler(ctx, intfr);
64 break;
65 case EXCEPTION_COPROCESSOR_SEGMENT_OVERRUN:
66 coprocessor_segment_overrun_handler(ctx, intfr);
67 break;
68 case EXCEPTION_SEGMENT_SELECTOR_NOTPRESENT:
69 segment_selector_not_present_handler(ctx, intfr);
70 break;
71 case EXCEPTION_INVALID_TSS:
72 invalidtss_handler(ctx, intfr);
73 break;
74 case EXCEPTION_GENERAL_PROTECTION_FAULT:
75 gpf_handler(ctx, intfr);
76 break;
77 case EXCEPTION_PAGE_FAULT:
78 pagefault_handler(ctx, intfr);
79 break;
80 case EXCEPTION_RESERVED:
81 // reserved, do not use.
82 break;
83 case EXCEPTION_FLOATING_POINT_ERROR:
84 fpu_handler(ctx, intfr);
85 break;
86 case EXCEPTION_ALIGNMENT_CHECK:
87 alignment_check_handler(ctx, intfr);
88 break;
89 case EXCEPTION_SEVERE_MACHINE_CHECK:
90 _MtSetIRQL(HIGH_LEVEL); // machine check, like NMI, high irql.
91 severe_machine_check_handler(ctx, intfr);
92 break;
93 case KEYBOARD_INTERRUPT:
94 MtRaiseIRQL(DIRQL_KEYBOARD, &oldIrql);
95 keyboard_handler();
96 MtLowerIRQL(oldIrql);
97 break;
98 case LAPIC_INTERRUPT:
99 MtRaiseIRQL(DIRQL_TIMER, &oldIrql);
100 lapic_handler(schedulerEnabled);
101 MtLowerIRQL(oldIrql);
102 break;
103 default:
104 gop_printf(0xFFFF0000, "Interrupt Exception: ");
105 gop_printf(0xFFFFFFFF, "%d\r\n", vec_num);
106 break;
107 }
108}

23static inline bool is_canonical_ptr(uint64_t x) {
24 // x86_64 canonical: bits 63..48 must be sign-extension of bit 47
25 uint64_t hi = x >> 47;
26 return (hi == 0 || hi == 0x1FFFF);
27}

29static void print_stack_trace(int depth) {
30 return;
31 uint64_t* rbp = (uint64_t*)__read_rbp();
32 for (int i = 0; rbp != NULL && i < depth; ++i) {
33 if (!MtIsAddressValid((void*)rbp)) break;
34 if (!MtIsAddressValid((void*)(rbp + 1))) break;
35
36 uint64_t saved_ret = *(rbp + 1);
37
38 /* Validate saved_ret before any symbolization/dereference */
39 if (!is_canonical_ptr(saved_ret)) break;
40 /* optional: ensure saved_ret is in known kernel text range if you have that info */
41 /* if (!address_in_kernel_text(saved_ret)) break; */
42
43 /* Print raw hex — avoid any logger symbolization that may dereference */
44 gop_printf(COLOR_ORANGE, "frame %d rbp=%016llx ret=%016llx\n",
45 i,
46 (unsigned long long)(uintptr_t)rbp,
47 (unsigned long long)saved_ret);
48
49 uint64_t next_rbp_val = *rbp;
50 if (next_rbp_val == 0) break;
51 if (!is_canonical_ptr(next_rbp_val)) break;
52
53 /* SANITY: next RBP should be at a *higher* address (older frame)
54 and not unreasonably far away. Enforce both direction and delta. */
55 uintptr_t cur = (uintptr_t)rbp;
56 uintptr_t next = (uintptr_t)next_rbp_val;
57 if (next <= cur) break;
58 if (next - cur > (16 * 1024 * 1024)) break; /* 16MB cap */
59
60 /* optional: require alignment */
61 if (next & 0xF) break;
62
63 rbp = (uint64_t*)(uintptr_t)next_rbp_val;
64 }
65}

68static void print_lastfunc_chain(uint32_t color) {
69 // Start at the oldest entry: that's the slot `index` points to (next write).
70 int idx = lastfunc_history.current_index;
71
72 int start = (idx + 1) % LASTFUNC_HISTORY_SIZE;
73 bool first = true;
74 for (int i = 0; i < LASTFUNC_HISTORY_SIZE; i++) { // start from oldest to newest
75 idx = (start + i) % LASTFUNC_HISTORY_SIZE;
76 char* name = (char*)lastfunc_history.names[idx];
77 if (!*name) break;
78 if (!first) {
79 gop_printf_forced(color, " -> ");
80 }
81 gop_printf_forced(color, "%s", name);
82 first = false;
83 }
84}

213void MtBugcheck(CTX_FRAME* context, INT_FRAME* int_frame, BUGCHECK_CODES err_code, uint32_t additional, bool isAdditionals) {
214 // Critical system error, instead of triple faulting, we hang the system with specified error codes.
215 // Disable interrupts if they werent disabled before.
216 __cli();
217 isBugChecking = true;
218 bool isThereIntFrame = (int_frame) ? true : false; // basic ternary
219#ifdef DEBUG
220 IRQL recordedIrql = cpu.currentIrql;
221#endif
222 // Force to be redrawn from the top, instead of last place.
223 cursor_x = 0;
224 cursor_y = 0;
225 _MtSetIRQL(HIGH_LEVEL); // SET the irql to high level (not raise) (we could raise, but this takes less cycles and so is faster) (When I will integrate multi core functionality, this should SetIRQL to each cpu core.
226
227 // Clear the screen to blue (bsod windows style)
228 gop_clear_screen(&gop_local, 0xFF0035b8);
229 // check if nullptr deref.
230 if (err_code == PAGE_FAULT && isAdditionals && additional == 0) {
231 err_code = NULL_POINTER_DEREFERENCE;
232 }
233 // Write some debugging and an error message
234 gop_printf_forced(0xFFFFFFFF, "FATAL ERROR: Your system has encountered a fatal error.\n\n");
235 gop_printf_forced(0xFFFFFFFF, "Your system has been stopped for safety.\n\n");
236 char* stopCodeToStr = ""; // empty at first.
237 resolveStopCode(&stopCodeToStr, err_code);
238 uint64_t rspIfExist;
239 if (context) {
240 rspIfExist = (context->rsp) ? context->rsp : (uint64_t)(-1);
241 }
242 gop_printf_forced(0xFFFFFFFF, "**STOP CODE: ");
243 gop_printf_forced(0xFF8B0000, "%s", stopCodeToStr);
244 gop_printf_forced(0xFF00FF00, " (numerical: %d)**", err_code);
245 if (context) {
246 gop_printf_forced(0xFFFFFFFF,
247 "\n\nRegisters:\n\n"
248 "RAX: %p RBX: %p RCX: %p RDX: %p\n\n"
249 "RSI: %p RDI: %p RBP: %p RSP: %p\n\n"
250 "R8 : %p R9 : %p R10: %p R11: %p \n\n"
251 "R12: %p R13: %p R14: %p R15: %p\n\n\n",
252 context->rax,
253 context->rbx,
254 context->rcx,
255 context->rdx,
256 context->rsi,
257 context->rdi,
258 context->rbp,
259 rspIfExist,
260 context->r8,
261 context->r9,
262 context->r10,
263 context->r11,
264 context->r12,
265 context->r13,
266 context->r14,
267 context->r15
268 );
269 }
270 else {
271 gop_printf_forced(0xFFFF0000, "\n\n\n**ERROR: NO REGISTERS.**\n");
272 }
273 // don't alert if there is no interrupt frame, the user shouldn't care and know. - i should do an IFDEF here for debug, but I could not remember that I didn't define, i'd rather keep it like this for now.
274 if (isThereIntFrame) {
275 gop_printf_forced((uint32_t)-1,
276 "Exceptions:\n\n"
277 "Vector Number: %d Error Code: %d\n\n"
278 "RIP: %p CS: %p RFLAGS: %b\n",
279 int_frame->vector,
280 int_frame->error_code,
281 int_frame->rip,
282 int_frame->cs,
283 int_frame->rflags
284 );
285 }
286#ifdef DEBUG
287 gop_printf_forced(0xFFFFA500, "**Last IRQL: %d**\n", recordedIrql);
288#endif
289 if (isAdditionals) {
290 if (err_code == PAGE_FAULT) {
291 gop_printf_forced(0xFFFFA500, "\n\n**FAULTY ADDRESS: %p**\n", additional);
292 }
293 else {
294 gop_printf_forced(0xFFBF40BF, "\n\n**ADDITIONALS: %p**\n", additional);
295 }
296 }
297#ifdef DEBUG
298 uint32_t currTid = (cpu.currentThread) ? cpu.currentThread->TID : (uint32_t)-1;
299 gop_printf_forced(0xFFFFFF00, "Current Thread ID: %d\n", currTid);
300#endif
301#ifdef DEBUG
302 if (lastfunc_history.names[lastfunc_history.current_index][0] != '\0') {
303 gop_printf_forced(0xFFBF40BF, "\n**FUNCTION TRACE (oldest to newest): ");
304 print_lastfunc_chain(0xFFBF40BF);
305 gop_printf_forced(0xFFBF40BF, "**");
306 }
307
308 // call stack trace
309 gop_printf(COLOR_GREEN, "\n\nCall Stack Trace:\n");
310 print_stack_trace(10); // 10 function calls;
311#endif
312 //test
313 update_pic_mask_for_current_irql();
314 __cli();
315 // spin the CPU.
316 while (1) {
317 NOTHING;
318 }
319}

21void MtQueueDPC(volatile DPC* dpc) {
22 tracelast_func("queue_dpc");
23 if (!dpc) return;
24 dpc->Next = NULL;
25
26 // Sorted insertion mechanism by priority (higher priority -> inserted to the head)
27 if (!dpcQueueHead) {
28 // Starting with empty queue.
29 dpcQueueHead = dpcQueueTail = dpc;
30 return;
31 }
32 // not an empty queue -> check priority.
33 // check if the priority is highest.
34 if (dpc->priority > dpcQueueHead->priority) {
35 // Insert at the front (head)
36 dpc->Next = dpcQueueHead;
37 dpcQueueHead = dpc;
38 return;
39 }
40 // else, find our insertion point.
41 volatile DPC* cur = dpcQueueHead;
42 // Check each DPC entry for it's priority, and insert the DPC requested accordingly.
43 while (cur->Next && cur->Next->priority >= dpc->priority) {
44 cur = cur->Next;
45 }
46 // Singular linked list.
47 dpc->Next = cur->Next;
48 cur->Next = dpc;
49 if (!dpc->Next) {
50 dpcQueueTail = dpc;
51 }
52}

54void RetireDPCs(void) {
55 tracelast_func("RetireDPCs");
56 if (!dpcQueueHead) return;
57
58 IRQL oldIrql;
59 uint64_t flags;
60
61 // 1) Raise once
62 MtRaiseIRQL(DISPATCH_LEVEL, &oldIrql);
63
64 // 2) Acquire lock for the whole drain
65 MtAcquireSpinlock(&dpc_lock, &flags);
66
67 // 3) Drain the queue
68 while (dpcQueueHead) {
69 volatile DPC* d = dpcQueueHead;
70 dpcQueueHead = d->Next;
71 if (!dpcQueueHead) {
72 dpcQueueTail = NULL;
73 }
74 // release lock so callback can queue new DPCs if needed
75 MtReleaseSpinlock(&dpc_lock, flags);
76
77 // STILL at DISPATCH_LEVEL
78 if (d->hasCtx) d->callbackWithCtx(d->ctx);
79 else if (d->callback) d->callback();
80
81 // re-acquire for next pop
82 MtAcquireSpinlock(&dpc_lock, &flags);
83 }
84
85 // 4) Release lock and lower once
86 MtReleaseSpinlock(&dpc_lock, flags);
87 MtLowerIRQL(oldIrql);
88}

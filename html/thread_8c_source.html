11// Call with freedTid == 0 ? allocate a new TID (returns 0 on failure)
12// Call with freedTid > 0 ? release that TID back into the pool (always returns 0)
14static uint32_t ManageTID(uint32_t freedTid)
15{
16 static uint32_t nextTID = MIN_TID;
17 static uint32_t freePool[MAX_FREE_POOL];
18 static uint32_t freeCount = 0;
19 uint32_t result = 0;
20
21 if (freedTid) {
22 // Release path: push into free pool if aligned & room
23 if ((freedTid % ALIGN_DELTA) == 0 && freeCount < MAX_FREE_POOL) {
24 freePool[freeCount++] = freedTid;
25 }
26 // else drop silently
27 }
28 else {
29 // Allocate path:
30 if (freeCount > 0) {
31 // Reuse most-recently freed
32 result = freePool[--freeCount];
33 }
34 else {
35 // Hand out next aligned TID
36 result = nextTID;
37 nextTID += ALIGN_DELTA;
38
39 // Wrap/overflow check
40 if (nextTID < ALIGN_DELTA || result > MAX_TID) {
41 // Exhausted all TIDs
42 result = 0;
43 }
44 }
45 }
46 return result;
47}

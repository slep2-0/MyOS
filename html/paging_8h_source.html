23// Flags for PDE/PTE
24// Included = BIT SET (1)
25// Not Included = BIT NOT SET (0)
26typedef enum _FLAGS {
27 PAGE_PRESENT = 1 << 0, // Bit 0
28 // 0 = page not present (access causes page fault)
29 // 1 = page is present, MMU translates virtual addresses
30
31 PAGE_RW = 1 << 1, // Bit 1
32 // 0 = read-only
33 // 1 = read/write
34
35 PAGE_USER = 1 << 2, // Bit 2
36 // 0 = supervisor (kernel) only
37 // 1 = user-mode access allowed
38
39 PAGE_PWT = 0x8, // Bit 3
40 // Page Write-Through
41 // 0 = write-back caching
42 // 1 = write-through caching
43
44 PAGE_PCD = 0x10, // Bit 4
45 // Page Cache Disable
46 // 0 = cacheable
47 // 1 = cache disabled
48
49 PAGE_ACCESSED = 0x20, // Bit 5
50 // Set by CPU when page is read or written
51
52 PAGE_DIRTY = 0x40, // Bit 6
53 // Set by CPU when page is written to
54
55 PAGE_PS = 0x80, // Bit 7
56 // Page Size
57 // 0 = normal 4KB page
58 // 1 = large page (4MB in PDE, 2MB in PTE for PAE/long mode)
59
60 PAGE_GLOBAL = 0x100, // Bit 8
61 // Global page
62 // Not flushed from TLB on CR3 reload
63} FLAGS;

92static inline void* MtTranslateKernelPhysicalToVirtual(uintptr_t phys) {
93 // assume PHYS_MEM_OFFSET and MEM_TO_MAP/UNMAPPED_LOW_MEM_SIZE are known
94 if (phys >= UNMAPPED_LOW_MEM_SIZE && phys < MEM_TO_MAP) {
95 return (void*)(phys + PHYS_MEM_OFFSET);
96 }
97 return NULL; // not mapped
98}

100static inline uintptr_t MtTranslateKernelVirtualToPhysical(void* v) {
101 uintptr_t va = (uintptr_t)v;
102 if (va >= PHYS_MEM_OFFSET && va < PHYS_MEM_OFFSET + MEM_TO_MAP) {
103 return va - PHYS_MEM_OFFSET;
104 }
105 return 0; // not in direct-map window
106}

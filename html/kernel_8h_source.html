77static int MemoryTest(void) {
78 void* blocks[ALLOCATIONS];
79
80 // Allocation + write test
81 for (int i = 0; i < ALLOCATIONS; i++) {
82 blocks[i] = MtAllocateVirtualMemory(BLOCK_SIZE, ALIGNMENT);
83 if (!blocks[i]) {
84 gop_printf_forced(0xFFFF0000, "Allocation failed at index %d\n", i);
85 return -1;
86 }
87
88 // Alignment test
89 if ((uintptr_t)blocks[i] % ALIGNMENT != 0) {
90 gop_printf_forced(0xFFFF8000, "Misaligned block at index %d: %p\n", i, blocks[i]);
91 return -2;
92 }
93
94 // Fill memory
95 for (int j = 0; j < BLOCK_SIZE; j++) {
96 ((uint8_t*)blocks[i])[j] = (uint8_t)(i + j);
97 }
98 }
99
100 // Verify and free
101 for (int i = 0; i < ALLOCATIONS; i++) {
102 for (int j = 0; j < BLOCK_SIZE; j++) {
103 if (((uint8_t*)blocks[i])[j] != (uint8_t)(i + j)) {
104 gop_printf_forced(0xFF0000FF, "Memory corruption at block %d, byte %d\n", i, j);
105 return -3;
106 }
107 }
108
109 MtFreeVirtualMemory(blocks[i]);
110 }
111
112 gop_printf_forced(0xFF00FF00, "Memory test completed successfully.\n");
113 return 0;
114}

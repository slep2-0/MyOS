18void queue_dpc(DPC* dpc) {
19 tracelast_func("queue_dpc");
20 if (!dpc) return;
21 dpc->Next = NULL;
22
23 // Sorted insertion mechanism by priority (higher priority -> inserted to the head)
24 if (!dpcQueueHead) {
25 // Starting with empty queue.
26 dpcQueueHead = dpcQueueTail = dpc;
27 return;
28 }
29 // not an empty queue -> check priority.
30 // check if the priority is highest.
31 if (dpc->priority > dpcQueueHead->priority) {
32 // Insert at the front (head)
33 dpc->Next = dpcQueueHead;
34 dpcQueueHead = dpc;
35 return;
36 }
37 // else, find our insertion point.
38 DPC* cur = dpcQueueHead;
39 // Check each DPC entry for it's priority, and insert the DPC requested accordingly.
40 while (cur->Next && cur->Next->priority >= dpc->priority) {
41 cur = cur->Next;
42 }
43 // Singular linked list.
44 dpc->Next = cur->Next;
45 cur->Next = dpc;
46 if (!dpc->Next) {
47 dpcQueueTail = dpc;
48 }
49}

53void DispatchDPC(void) {
54 tracelast_func("DispatchDPC");
55 if (!dpcQueueHead) return;
56
57 IRQL oldIrql;
58 RaiseIRQL(DISPATCH_LEVEL, &oldIrql);
59
60 // This loop will now complete, because TimerDPC no longer hijacks execution.
61 while (dpcQueueHead) {
62 DPC* d = dpcQueueHead;
63 dpcQueueHead = d->Next;
64 if (!dpcQueueHead) {
65 dpcQueueTail = NULL;
66 }
67 if (d->callback) {
68 if (d->hasCtx) {
69 d->callbackWithCtx(d->ctx);
70 }
71 else {
72 d->callback();
73 }
74 }
75 }
76
77 // Lower the IRQL *before* checking the reschedule flag.
78 LowerIRQL(oldIrql);
79
80 // Now that we are back at a safe IRQL (PASSIVE_LEVEL\DISPATCH_LEVEL), check if we need to schedule.
81 if (reschedule_needed) {
82 reschedule_needed = false; // Clear the flag
83 Yield(); // Yield() is just a clean wrapper for Schedule()
84 }
85}

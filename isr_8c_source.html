24void isr_handler64(int vec_num, INTERRUPT_FULL_REGS* r) {
25 IRQL oldIrql;
26 CTX_FRAME ctx;
27 INT_FRAME intfr;
28
29 // Since we are now using both a registration context, and an interrupt context, for compatibility I didn't change the asm stub, and just created 3 structs, 2 are the ctx and int, third is both (legacy).
30 ctx.rax = r->rax;
31 ctx.rbx = r->rbx;
32 ctx.rcx = r->rcx;
33 ctx.rdi = r->rdi;
34 ctx.rsi = r->rsi;
35 ctx.rsp = (uint64_t)-1; // max val, to signify invalidation that it's not here.
36 ctx.r8 = r->r8;
37 ctx.r9 = r->r9;
38 ctx.r10 = r->r10;
39 ctx.r11 = r->r11;
40 ctx.r12 = r->r12;
41 ctx.r13 = r->r13;
42 ctx.r14 = r->r14;
43 ctx.r15 = r->r15;
44
45 intfr.cs = r->cs;
46 intfr.error_code = r->error_code;
47 intfr.rflags = r->rflags;
48 intfr.rip = r->rip;
49 intfr.vector = r->vector;
50
51 switch (vec_num) {
52 case EXCEPTION_DIVIDE_BY_ZERO:
53 dividebyzero_handler(&ctx, &intfr);
54 break;
55 case EXCEPTION_SINGLE_STEP:
56 debugsinglestep_handler(&ctx, &intfr);
57 break;
58 case EXCEPTION_NON_MASKABLE_INTERRUPT:
59 _SetIRQL(HIGH_LEVEL); // Non Maskable Interrupt - basically when the CPU encounters a hardware fault, cannot be masked, very alarming.
60 nmi_handler(&ctx, &intfr);
61 break;
62 case EXCEPTION_BREAKPOINT:
63 breakpoint_handler(&ctx, &intfr);
64 break;
65 case EXCEPTION_OVERFLOW:
66 overflow_handler(&ctx, &intfr);
67 break;
68 case EXCEPTION_BOUNDS_CHECK:
69 boundscheck_handler(&ctx, &intfr);
70 break;
71 case EXCEPTION_INVALID_OPCODE:
72 invalidopcode_handler(&ctx, &intfr);
73 break;
74 case EXCEPTION_NO_COPROCESSOR:
75 nocoprocessor_handler(&ctx, &intfr);
76 break;
77 case EXCEPTION_DOUBLE_FAULT:
78 doublefault_handler(&ctx, &intfr);
79 break;
80 case EXCEPTION_COPROCESSOR_SEGMENT_OVERRUN:
81 coprocessor_segment_overrun_handler(&ctx, &intfr);
82 break;
83 case EXCEPTION_SEGMENT_SELECTOR_NOTPRESENT:
84 segment_selector_not_present_handler(&ctx, &intfr);
85 break;
86 case EXCEPTION_INVALID_TSS:
87 invalidtss_handler(&ctx, &intfr);
88 break;
89 case EXCEPTION_GENERAL_PROTECTION_FAULT:
90 gpf_handler(&ctx, &intfr);
91 break;
92 case EXCEPTION_PAGE_FAULT:
93 pagefault_handler(&ctx, &intfr);
94 break;
95 case EXCEPTION_RESERVED:
96 // reserved, do not use.
97 break;
98 case EXCEPTION_FLOATING_POINT_ERROR:
99 fpu_handler(&ctx, &intfr);
100 break;
101 case EXCEPTION_ALIGNMENT_CHECK:
102 alignment_check_handler(&ctx, &intfr);
103 break;
104 case EXCEPTION_SEVERE_MACHINE_CHECK:
105 _SetIRQL(HIGH_LEVEL); // machine check, like NMI, high irql.
106 severe_machine_check_handler(&ctx, &intfr);
107 break;
108 case TIMER_INTERRUPT:
109 RaiseIRQL(CLOCK_LEVEL, &oldIrql);
110 timer_handler();
111 LowerIRQL(oldIrql);
112 break;
113 case KEYBOARD_INTERRUPT:
114 RaiseIRQL(DIRQL_KEYBOARD, &oldIrql);
115 keyboard_handler();
116 LowerIRQL(oldIrql);
117 break;
118 case ATA_INTERRUPT:
119 RaiseIRQL(DIRQL_PRIMARY_ATA, &oldIrql);
120 ata_handler();
121 LowerIRQL(oldIrql);
122 break;
123 default:
124 gop_printf(&gop_local, 0xFFFF0000, "Interrupt Exception: ");
125 gop_printf(&gop_local, 0xFFFFFFFF, "%d\r\n", vec_num);
126 break;
127 }
128}

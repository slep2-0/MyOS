20// PIO Read one sector from LBA into buf.
21static bool ata_read_sector(BLOCK_DEVICE* dev, uint32_t lba, void* buf) {
22 UNREFERENCED_PARAMETER(dev);
23 tracelast_func("ata_read_sector");
24 // Wait until BSY is 0.
25 while (__inbyte(ATA_PRIMARY_BASE + 7) & 0x80) {/*none*/}
26
27 // select drive and head, send LBA.
28 __outbyte(ATA_PRIMARY_BASE + 6, (unsigned char)(0xE0 | ((lba >> 24) & 0x0F)));
29 __outbyte(ATA_PRIMARY_BASE + 2, 1); // sector count = 1.
30 __outbyte(ATA_PRIMARY_BASE + 3, (uint8_t)lba);
31 __outbyte(ATA_PRIMARY_BASE + 4, (uint8_t)(lba >> 8));
32 __outbyte(ATA_PRIMARY_BASE + 5, (uint8_t)(lba >> 16));
33 __outbyte(ATA_PRIMARY_BASE + 7, 0x20); // cmd: read
34
35 // Wait for BSY = 0 again (after sending the command)
36 gop_printf(&gop_local, 0xEED3D3D3, "Waiting for BSY=0 (READ)\n");
37 while (__inbyte(ATA_PRIMARY_BASE + 7) & 0x80) {}
38
39 // Wait DRQ
40 gop_printf(&gop_local, 0xEED3D3D3, "Waiting for DRQ=1 (READ)\n");
41 while (!(__inbyte(ATA_PRIMARY_BASE + 7) & 0x08)) {/*wait*/}
42
43 // Read 256 * 16 bit words.
44 uint16_t* ptr = buf;
45 gop_printf(&gop_local, 0xEED3D3D3, "Reading data now...\n");
46 for (int i = 0; i < 256; i++) {
47 ptr[i] = __inword(ATA_PRIMARY_BASE);
48 }
49 return true;
50}

52// PIO Write one sector from buf to LBA
53static bool ata_write_sector(BLOCK_DEVICE* dev, uint32_t lba, const void* buf) {
54 UNREFERENCED_PARAMETER(dev);
55 tracelast_func("ata_write_sector");
56 // Busy wait.
57 while (__inbyte(ATA_PRIMARY_BASE + 7) & 0x80) {}
58
59 // Again, OUT to the primary base, this time for cmd write.
60 __outbyte(ATA_PRIMARY_BASE + 6, (unsigned char)(0xE0 | ((lba >> 24) & 0x0F)));
61 __outbyte(ATA_PRIMARY_BASE + 2, 1);
62 __outbyte(ATA_PRIMARY_BASE + 3, (uint8_t)lba);
63 __outbyte(ATA_PRIMARY_BASE + 4, (uint8_t)(lba >> 8));
64 __outbyte(ATA_PRIMARY_BASE + 5, (uint8_t)(lba >> 16));
65 __outbyte(ATA_PRIMARY_BASE + 7, 0x30); // cmd: write
66
67
68 // Wait for BSY = 0 again (after sending the command)
69 gop_printf(&gop_local, 0xEED3D3D3, "Waiting for BSY=0 (WRITE)\n");
70 while (__inbyte(ATA_PRIMARY_BASE + 7) & 0x80) {}
71
72 gop_printf(&gop_local, 0xEED3D3D3, "Waiting for DRQ=1 (WRITE)\n");
73 while (!(__inbyte(ATA_PRIMARY_BASE + 7) & 0x08)) {}
74
75 const uint16_t* ptr = buf;
76 gop_printf(&gop_local, 0xEED3D3D3, "Writing data now...\n");
77 for (int i = 0; i < 256; i++) {
78 __outword(ATA_PRIMARY_BASE, ptr[i]);
79 }
80 return true;
81}

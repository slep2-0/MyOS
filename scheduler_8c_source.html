24void InitScheduler(void) {
25 tracelast_func("InitScheduler");
26 cpu.schedulerEnabled = true;
27
28 // --- Build a CLEAN context for the idle thread from scratch ---
29 // Do NOT call save_context().
30
31 CTX_FRAME cfm;
32 kmemset(&cfm, 0, sizeof(cfm)); // Start with a clean, all-zero context
33
34 // Set only the essential registers for starting the thread
35 cfm.rsp = (uint64_t)(idleStack + IDLE_STACK_SIZE);
36 cfm.rip = (uint64_t)kernel_idle_checks;
37
38 // Assign the clean context to the idle thread
39 idleThread.registers = cfm;
40 idleThread.threadState = RUNNING;
41 idleThread.nextThread = NULL;
42
43 // Set the idle thread as the one currently running
44 cpu.currentThread = &idleThread;
45
46 // The ready queue starts empty
47 cpu.readyQueue.head = cpu.readyQueue.tail = NULL;
48}

50// Enqueue the thread if it's still RUNNABLE.
51static void enqueue_runnable(Thread* t) {
52 tracelast_func("enqueue_runnable");
53 if (t->threadState == RUNNING) {
54 t->threadState = READY;
55 enqueue(&cpu.readyQueue, t); // Insert into CPU ready queue
56 }

59void Schedule(void) {
60 tracelast_func("Schedule");
61 if (isScheduleDpcQueued) {
62 isScheduleDpcQueued = false;
63 }
64 IRQL oldIrql;
65 RaiseIRQL(DISPATCH_LEVEL, &oldIrql);
66
67 Thread* prev = cpu.currentThread;
68 if (prev != &idleThread) {
69 save_context(&prev->registers);
70 // Only enqueue non-idle
71 enqueue_runnable(prev);
72 }
73
74 Thread* next = dequeue(&cpu.readyQueue);
75 if (!next) {
76 next = &idleThread;
77 }
78
79 next->threadState = RUNNING;
80 cpu.currentThread = next;
81 LowerIRQL(oldIrql);
82 restore_context(&next->registers);
83}

101static inline void mask_irq(PIC_IRQ_LINE irq_line) {
102 uint16_t port;
103 uint8_t mask;
104 if (irq_line < 8) {
105 port = PIC1_DATA_MASTER;
106 }
107 else {
108 port = PIC2_DATA_SLAVE;
109 irq_line -= 8;
110 }
111 mask = __inbyte(port);
112 mask |= (1 << irq_line);
113 __outbyte(port, mask);
114}

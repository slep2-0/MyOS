19// Scheduling disabling is by flipping a global flag, I should make a CPU structure that is the current CPU states and data.
20// Interrupts are disabled based on the IRQL <=, so even at the IRQL level they are in they are disabled.
21typedef enum _IRQL {
22 PASSIVE_LEVEL = 0, // Normal thread execution, all interrupts enabled.
23 DISPATCH_LEVEL = 2, // Scheduler disabled, page faults fatal. (page faults are always fatal for now, until I implement an exception handler)
24 // Device DIRQLs (chosen so IRQn + DIRQL = PROFILE_LEVEL (27))
25 DIRQL_SECONDARY_ATA = 12, // IRQ15 – Secondary ATA Channel
26 DIRQL_PRIMARY_ATA = 13, // IRQ14 – Primary ATA Channel
27 DIRQL_FPU = 14, // IRQ13 – FPU / Coprocessor
28 DIRQL_MOUSE = 15, // IRQ12 – Mouse
29 DIRQL_PERIPHERAL11 = 16, // IRQ11 – Free for peripherals
30 DIRQL_PERIPHERAL10 = 17, // IRQ10 – Free for peripherals
31 DIRQL_PERIPHERAL9 = 18, // IRQ9 – Free / redirected cascade
32 DIRQL_RTC = 19, // IRQ8 – RTC / CMOS Alarm
33 DIRQL_LPT1 = 20, // IRQ7 – LPT1 / Printer
34 DIRQL_FLOPPY = 21, // IRQ6 – Floppy Disk
35 DIRQL_SOUND_LPT2 = 22, // IRQ5 – Sound Card / LPT2
36 DIRQL_COM1 = 23, // IRQ4 – Serial COM1
37 DIRQL_COM2 = 24, // IRQ3 – Serial COM2
38 DIRQL_CASCADE = 25, // IRQ2 – Cascade (IRQs 8–15)
39 DIRQL_KEYBOARD = 26, // IRQ1 – Keyboard
40 DIRQL_TIMER = 27, // IRQ0 – System Timer
41 PROFILE_LEVEL = 27, // Profile timer interrupts (alias of DIRQL_TIMER)
42 CLOCK_LEVEL = 28, // Clock/timer interrupts (second-level timer) (actual clock IRQ timer, for scheduler, time-of-day clock, and general timers, even context switching)
43 SYNCH_LEVEL = 29, // Synchronization level (internal kernel use) (unused in my kernel (until SMP), this level is used for multi core CPU synchronization)
44 POWER_LEVEL = 30, // Power failure interrupts
45 HIGH_LEVEL = 31 // NMI and machine‑check (non‑maskable) (also gets set at bugchecks, to mask all interrupts)
46} IRQL;

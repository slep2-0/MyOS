36static bool grow_heap_by_one_page(void) {
37 tracelast_func("grow_heap_by_one_page");
38 enforce_max_irql(DISPATCH_LEVEL);
39 // grab a physical frame

43 // map it at the end of the heap.

47 // Zero the page.
48 kmemset((void*)heap_current_end, 0, FRAME_SIZE);
49 // insert a new region

51 block->size = FRAME_SIZE;
52 block->next = free_list;
53 free_list = block;
54 coalesce_free_list();
55
56 // advance our end.

61static void insert_block_sorted(BLOCK_HEADER* newblock) {
62 tracelast_func("insert_block_sorted");
63 enforce_max_irql(DISPATCH_LEVEL);
64 if (!free_list || newblock < free_list) {
65 newblock->next = free_list;
66 free_list = newblock;
67 return;
68 }
69
70 BLOCK_HEADER* current = free_list;
71 while (current->next && current->next < newblock) {
72 current = current->next;
73 }
74
75 newblock->next = current->next;
76 current->next = newblock;
77}

115/* Allocate `size` bytes, aligned to `align` bytes */
116// IRQL: Passive level minimum (for all memory operations, unless expansion)
117void* kmalloc(size_t wanted_size, size_t align) {
118 tracelast_func("kmalloc");
119 enforce_max_irql(DISPATCH_LEVEL);
120 /* Round up the requested size to satisfy alignment of payload */
121 size_t payload_size = (wanted_size + align - 1) & ~(align - 1);
122 size_t total_size = payload_size + sizeof(BLOCK_HEADER);
123
124 /* First‑fit search through free list */
125 // A pointer to a pointer, since we also want the previous node (which is cur, a ponter to the next field of the previous node.)
126 BLOCK_HEADER** cur = &free_list;
127 while (*cur) {
128 BLOCK_HEADER* blk = *cur;
129
130 if (blk->size >= total_size) {
131 /* If the block is big enough to split */
132 if (blk->size >= total_size + sizeof(BLOCK_HEADER) + align) {
133 /* Split off the tail into a new free block */
134 BLOCK_HEADER* next_blk = (BLOCK_HEADER*)((uintptr_t)blk + total_size);
135 next_blk->size = blk->size - total_size;
136 next_blk->next = blk->next;
137
138 /* Shrink current block to allocated size */
139 blk->size = total_size;
140 *cur = next_blk;
141 }
142 else {
143 /* Use the entire block */
144 *cur = blk->next;
145 }
146
147 void* raw_ptr = (void*)(blk + 1);
148 void* aligned_ptr = align_up((void*)((uintptr_t)raw_ptr + sizeof(void*)), align);
149 ((BLOCK_HEADER**)aligned_ptr)[-1] = blk;
150 return aligned_ptr;
151 }
152
153 cur = &blk->next;
154 }
155 if (!*cur) {
156 // Out of memory, add new page maps.
157 if (heap_current_end + FRAME_SIZE <= HEAP_END && grow_heap_by_one_page()) {
158 // new block now, restart the kmalloc.
159 kmalloc(wanted_size, align);
160 }
161 }
162 // No physical memory left.
163 return NULL;
164}

166/* Merge adjacent free blocks to reduce fragmentation */
167void coalesce_free_list(void) {
168 tracelast_func("coalesce_free_list");
169 enforce_max_irql(DISPATCH_LEVEL);
170 BLOCK_HEADER* b = free_list;
171 while (b && b->next) {
172 uintptr_t end_of_b = (uintptr_t)b + b->size;
173 if (end_of_b == (uintptr_t)b->next) {
174 /* Adjacent: merge */
175 b->size += b->next->size;
176 b->next = b->next->next;
177 }
178 else {
179 b = b->next;
180 }
181 }
182}

184/* Return a block to the free list */
185void kfree(void* ptr) {
186 tracelast_func("kfree");
187 enforce_max_irql(DISPATCH_LEVEL);
188 if (!ptr) {
189#ifdef DEBUG
190 gop_printf(&gop_local, 0xFFFF0000, "<-- KFREE() DEBUG --> nullptr passed as argument, returning\n");
191#endif
192 return;
193 }
194 /* Get header */
195 BLOCK_HEADER* blk = ((BLOCK_HEADER**)ptr)[-1];
196
197#ifdef DEBUG
198 gop_printf(&gop_local, 0xFFFFFF00, "<-- KFREE() DEBUG --> blk: %p\n", blk);
199#endif
200 /* Zero it out. */
201 // FATAL ERROR BEFORE, it zeroed out the block metadata as well, so kmalloc didn't even know it existed.
202 // The fix is to skip over the metadata, just zero the payload (actual data) (minus the actual block value, so we don't zero out the next 8 bytes in the next block)
203 // REMEMBER TO MYSELF: pointer arithmetic in C and C++ moves the pointer BASED ON THE TYPE OF IT! (so void ptr in 32 bit is move 4 bytes, or char which is move 1 byte, and here BLOCK_HEADER is 8 bytes, (and technically we just want to move 1 block header so we skip over it's metadata))
204 kmemset((void*)(blk + 1), 0, blk->size - sizeof(BLOCK_HEADER));
205 /* Push it onto the free list */
206 insert_block_sorted(blk);
207#ifdef DEBUG
208 gop_printf(&gop_local, 0xFFFFFF00 ,"<-- KFREE() DEBUG --> ");
209 gop_printf(&gop_local, 0xFF00FFFF, "Pushed the block to the free list.\n");
210#endif
211 /* Optionally merge neighbors */
212 coalesce_free_list();
213}

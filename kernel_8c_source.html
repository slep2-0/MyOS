32void copy_memory_map(BOOT_INFO* boot_info) {
33 if (!boot_info || !boot_info->MemoryMap) return;
34 if (boot_info->MapSize > MAX_MEMORY_MAP_SIZE) {
35 // handle error, memory map too big
36 bugcheck_system(NULL, NULL, MEMORY_MAP_SIZE_OVERRUN, 0, false);
37 }
38
39 // Copy the entire memory map into the static buffer
40 kmemcpy(memory_map_copy, boot_info->MemoryMap, boot_info->MapSize);
41
42 boot_info_local.MemoryMap = memory_map_copy;
43 boot_info_local.MapSize = boot_info->MapSize;
44 boot_info_local.DescriptorSize = boot_info->DescriptorSize;
45 boot_info_local.DescriptorVersion = boot_info->DescriptorVersion;
46}

48void copy_gop(BOOT_INFO* boot_info) {
49 if (!boot_info || !boot_info->Gop) return;
50
51 // Copy the GOP data to a local global variable

54 // Update all relevant pointers to point to the local copy

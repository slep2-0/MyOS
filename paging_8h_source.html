27typedef enum _FLAGS {
28 PAGE_PRESENT = 1 << 0, // 0 - ANY access to this page will fault, unmapped. 1 - the page is valid and the MMU will translate and allow accesses (subject to RW and USER)
29 PAGE_RW = 1 << 1, // 0 - page is read only, 1- page is read & write
30 PAGE_USER = 1 << 2, // 0 - supervisor only (kernel), user mode access will cause a fault. 1 - any access (both user mode and kernel mode) is allowed. (IF THIS IS SET, USER-MOED IS ALLOWED)
31 PAGE_PWT = 0x8,
32 PAGE_PCD = 0x10,
33 PAGE_ACCESSED = 0x20,
34 PAGE_DIRTY = 0x40,
35 PAGE_PS = 0x80,

37} FLAGS;

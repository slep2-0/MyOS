22void update_pic_mask_for_current_irql(void) {
23 tracelast_func("update_pic_mask_for_current_irql");
24 IRQL level = cpu.currentIrql;
25
26 // Mask any IRQ whose assigned IRQL is <= the current CPU IRQL.
27 // Unmask any IRQ whose assigned IRQL is > the current CPU IRQL.
28 for (uint8_t i = 0; i < IRQ_LINES; i++) {
29 if (irq_irql[i] <= level) {
30 mask_irq(i);
31 }
32 else {
33 unmask_irq(i);
34 }
35 }
36}

43void RaiseIRQL(IRQL new_irql, IRQL* old_irql) {
44 tracelast_func("RaiseIRQL");
45
46 // It's good practice to disable interrupts for the duration of the change.
47 // The CPU automatically does this when entering an ISR.
48 // We do it here to ensure atomicity.
49 __cli();
50
51 if (old_irql) {
52 *old_irql = cpu.currentIrql;
53 }
54
55 if (new_irql < cpu.currentIrql) {
56 // You cannot "raise" to a lower level. This is a fatal kernel bug.
57 bugcheck_system(NULL, NULL, IRQL_NOT_LESS_OR_EQUAL, 0, false);
58 }
59
60 cpu.currentIrql = new_irql;
61 update_pic_mask_for_current_irql();
62
63 // We do NOT call __sti() here. We let the caller (or iretq) handle it.
64 // In the case of Schedule(), interrupts are already enabled, so we restore that state.
65 __sti();
66}

68void LowerIRQL(IRQL new_irql) {
69 tracelast_func("LowerIRQL");
70
71 // Atomically update the IRQL and PIC mask.
72 __cli();
73
74 if (new_irql > cpu.currentIrql) {
75 // You cannot "lower" to a higher level. This is a fatal kernel bug.
76 bugcheck_system(NULL, NULL, IRQL_NOT_LESS_OR_EQUAL, 0, false);
77 }
78
79 cpu.currentIrql = new_irql;
80 update_pic_mask_for_current_irql();
81
82 // Now that the masks are correctly set for the lower IRQL,
83 // we can safely re-enable interrupts.
84 __sti();
85}
